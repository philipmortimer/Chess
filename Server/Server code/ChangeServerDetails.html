163     /**
164      * Gets the port number inputted and if it is an integer greater than or equal to 1, it is stored in the correct file and the form is updated.
165      * @param evt The save details button being pressed by the user
166      */
167     private void saveDetailsBtnActionPerformed(java.awt.event.ActionEvent evt) {                                               
168         try{
169             int portNum = Integer.parseInt(portNumberTxt.getText());
170             if(portNum<=0){//range check
171                 JOptionPane.showMessageDialog(this, "The port number must be an integer that is greater than 0.","Invalid Input",JOptionPane.OK_OPTION);
172                 return;
173             }
174             setPortNumberTextFile(portNum);
175             isPortNoBeingEdited=!isPortNoBeingEdited;
176             setComponentsBasedOnIsBeingEdited();loadDetailsFromFileOntoForm();
177         }catch(NumberFormatException e){
178             JOptionPane.showMessageDialog(this, "The port number must be an integer that is greater than 0.","Invalid Input",JOptionPane.OK_OPTION);
179             return;
180         }
181     }

220     /*
221     Loads details stored in the server settings text file and displays them on the form
222     */
223     private void loadDetailsFromFileOntoForm(){
224         try{
225             FileReader read = new FileReader(SERVER_SETTINGS_TEXT_FILE_NAME);
226             BufferedReader buffRead = new BufferedReader(read);
227             portNumberTxt.setText(buffRead.readLine());read.close();buffRead.close();
228         }catch(IOException e){
229             JOptionPane.showMessageDialog(this, "An error occured when reading from file: "+SERVER_SETTINGS_TEXT_FILE_NAME+". Error details "+e, "Error", JOptionPane.OK_OPTION);
230         }
231     }
232     /**
233      * Stores the new port number in the correct text file
234      * @param portNumber the new port number
235      */
236     private void setPortNumberTextFile(int portNumber){
237         try{
238             FileWriter write = new FileWriter(SERVER_SETTINGS_TEXT_FILE_NAME,false);
239             BufferedWriter buffWrite = new BufferedWriter(write);
240             buffWrite.write(String.valueOf(portNumber));
241             buffWrite.flush();write.flush();
242             buffWrite.close();write.close();
243         }catch(IOException e){
244             JOptionPane.showMessageDialog(this, "An error occured when writing to file: "+SERVER_SETTINGS_TEXT_FILE_NAME+". Error details "+e, "Error", JOptionPane.OK_OPTION);           
245         }
