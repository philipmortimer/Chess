 54     private void addMessageToTextArea(String message){
 55         DateTimeFormatter dtf = DateTimeFormatter.ofPattern("yyyy/MM/dd HH:mm:ss");  
 56         LocalDateTime now = LocalDateTime.now();

 58         //adds the message and time of the message

 61         }else{
 62             serverInfoTxt.setText(currentText+System.lineSeparator()+dtf.format(now)+" - "+message);
 63         }

207     private void changeServerStateBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                     
208         isServerRunning=!isServerRunning;//changes the sever state
209         if(isServerRunning){
210             try{
211                 //creates a new server and starts it
212                 errorMessagesArray=new String[0];
213                 serv = new Server(errorMessagesArray);
214                 serverThread = new Thread(serv);
215                 serverThread.start();//starts the server
216                 timer = new Timer(1000, new ActionListener() {//checks periodically to see if any erros have occured, and kills the server if this is the case
217                     @Override
218                     public void actionPerformed(ActionEvent e) {
219                         if(serv.haveAnyErrorsOccured()){
220                             //shuts the server down in any errros have occurred and displays these messages on the text field
221                             serverThread.interrupt();
222                             errorMessagesArray=serv.getErrorMessages();
223                             for(int i=0;i<errorMessagesArray.length;i++){
224                                 addMessageToTextArea(errorMessagesArray[i]);
225                             }
226                             try {
227                                 serv.closeServerSocket();
228                             } catch (IOException ex) {
229                                 addMessageToTextArea(ex.getMessage());
230                             }
231                             isServerRunning=false;
232                             serv =null;
233                             serverThread=null;
234                             timer.stop();
235                             updateCompontentsBasedOnServerStatus();
236                         }
237                     }
238                 });
239                 timer.start();//starts the timer
240             }catch(Exception e){
241                 isServerRunning=false;
242                 addMessageToTextArea(e.getMessage());
243                 updateCompontentsBasedOnServerStatus();
244             }
245         }else{
246             if(timer!=null){//stops the timer
247                 timer.stop();
248             }
249             try {
250                 serv.closeServerSocket();//closes the server safely
251             } catch (IOException ex) {
252                 addMessageToTextArea(ex.getMessage());
253             }
254             serv=null;serverThread=null;timer=null;
255         }
256         updateCompontentsBasedOnServerStatus();//updates the form
257     }

259     private void appointLeaderBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                 
260         // sets a new club leader if appropriate. This can also be used to choose a leader when no leaders exist
261         int confirm = JOptionPane.showConfirmDialog(this, "Are you would like to appoint a leader? If a leader exists, they will be replaced.", "Proceed?", JOptionPane.YES_NO_CANCEL_OPTION);
262         if(confirm==JOptionPane.YES_OPTION){
263             //appoints somebody else as the chess club leader
264             String newLeader = JOptionPane.showInputDialog(this,"Please input the username of the new leader:");
265             if(newLeader==null){
266                 return;
267             }
268             try{
269                 serv.queue.setUserAsLeaderAndMember(newLeader);
270                 JOptionPane.showMessageDialog(this,"The leader was appointed.","Leader Appointed.",JOptionPane.OK_OPTION);
271             }catch(Exception e){
272                 JOptionPane.showMessageDialog(this,"The username inputted was not valid and hence leadership has not changed.","Username invalid.",JOptionPane.OK_OPTION);    
273             }
274         }
