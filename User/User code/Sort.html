 14     * Sorts data using insertion sort
 15     * @param unsortedData a 2d string array of the unsorted data with each row representing one record and the nought element containing a String primary key (which is the field that will be used to sort)
 16     */
 17     public static void insertionSort(String[][]unsortedData){
 18         int indexToBeChecked=0;//loop variable that needs intialisation for method wide access
 19         String tempRecord[]=new String[unsortedData[0].length];//used as buffer storage of a record
 20         for(int indexOfElementToBeInserted=1;indexOfElementToBeInserted<unsortedData.length;indexOfElementToBeInserted++){
 21             for(int recordIndex=0;recordIndex<tempRecord.length;recordIndex++){//stores value of record to be "inserted"
 22                 tempRecord[recordIndex]=unsortedData[indexOfElementToBeInserted][recordIndex];
 23             }
 24             for(indexToBeChecked=indexOfElementToBeInserted-1;indexToBeChecked>=0&&(tempRecord[0].compareTo(unsortedData[indexToBeChecked][0])<0);indexToBeChecked--){
 25                 for(int recordIndex=0;recordIndex<tempRecord.length;recordIndex++){
 26                     unsortedData[indexToBeChecked+1][recordIndex]=unsortedData[indexToBeChecked][recordIndex];
 27                 }
 28             }
 29             for(int recordIndex=0;recordIndex<tempRecord.length;recordIndex++){
 30                 unsortedData[indexToBeChecked+1][recordIndex]=tempRecord[recordIndex];
 31             }
 32         }
 33     }
 34         /**
 35      * This function recursively splits all record down into sub arrays until each sub array only contains one record
 36      * these split arrays are merged back together using the merge function.
 37      * Therefore the data originally input into this function will be sorted (as arrays are passed by reference in java). The return value of this function is also the
 38      * sorted data.
 39      * @param data The data you want sorted. Assuming each row is one record and the first element in each row is a string that is the primary key.
 40      * @return The sorted data is returned with the data being sorted in ascending order (one row representing each record and the first element of each row containing the integer primary key).
 41      */
 42     public static String[][] mergeSort(String[][]data){
 43         if(data.length==1){
 44             return data;
 45         }
 46         int middle = data.length/2; //calculates the middle of the current data
 47         String[][]left=new String[middle][data[0].length];//stores the left half of the current data
 48         String[][]right = new String[data.length-middle][data[0].length];//stores the right half of the current data
 49         for(int yVal=0;yVal<data.length;yVal++){//fills up the left and right arrays so that the left array stores the left/ top half of the data array and the right array stores the remianing records
 50             if(yVal>=middle){
 51                 for(int xVal=0;xVal<data[0].length;xVal++){
 52                     right[yVal-middle][xVal]=data[yVal][xVal];
 53                 }
 54             }else{
 55                 for(int xVal=0;xVal<data[0].length;xVal++){
 56                     left[yVal][xVal]=data[yVal][xVal];
 57                 }
 58             }
 59         }
 60         return merge(mergeSort(left),mergeSort(right),data);//returns sorted data. calls the merge function on the data that has been recursively split into single elements
 61     }
 62     /**
 63      * 
 64      * @param left the left hand sub array of data
 65      * @param right the right hand sub array of data
 66      * @param dataToBeSorted the data to be sorted
 67      * @return returns the merged / sorted data 
 68      */
 69     private static String[][] merge(String[][]left,String[][]right,String dataToBeSorted[][]){
 70         int leftIndex=0;//stores index for left array
 71         int rightIndex=0;//stores index for right array
 72         int dataToBeStoredIndex=0;//stores the row index for the data to be sorted array
 73         while(leftIndex<left.length && rightIndex<right.length){//both righ and left arrays are already sorted. sorts one entire sub array and most of the other sub array
 74             if(left[leftIndex][0].compareTo(right[rightIndex][0])<0){
 75                 for(int xVal=0;xVal<dataToBeSorted[0].length;xVal++){
 76                     dataToBeSorted[dataToBeStoredIndex][xVal]=left[leftIndex][xVal];
 77                 }
 78                 dataToBeStoredIndex++;
 79                 leftIndex++;
 80             }else{
 81                 for(int xVal=0;xVal<dataToBeSorted[0].length;xVal++){
 82                     dataToBeSorted[dataToBeStoredIndex][xVal]=right[rightIndex][xVal];
 83                 }
 84                 dataToBeStoredIndex++;
 85                 rightIndex++;
 86             }
 87         }
 88         //as one of the sub arrays has been completely sorted, the remaining elements of the other array (which are already sorted) are appended onto the end of the array
 89         while(leftIndex<left.length){
 90             for(int xVal=0;xVal<dataToBeSorted[0].length;xVal++){
 91                 dataToBeSorted[dataToBeStoredIndex][xVal]=left[leftIndex][xVal];
 92             }
 93             dataToBeStoredIndex++;
 94             leftIndex++;
 95         }
 96         while(rightIndex<right.length){
 97             for(int xVal=0;xVal<dataToBeSorted[0].length;xVal++){
 98                 dataToBeSorted[dataToBeStoredIndex][xVal]=right[rightIndex][xVal];
 99             }
100             dataToBeStoredIndex++;
101             rightIndex++;
102         }
103         return dataToBeSorted;
104     }

112     /**
113      * Performs a quick sort assuming that each record in the array contains an integer value at the nought element.
114      * @param first A parameter used during quicksort
115      * @param last A parameter used during quicksort
116      * @param arr The array to sort
117      */
118     private static void quickSortInt(int first, int last,String arr[][]){
119         int left, right, pivot;
120         String buffer[]=new String[arr[0].length];
121         left =first; right=last;
122         pivot = Integer.parseInt(arr[(left+right)/2][0]);
123         while(left<=right){
124             while(Integer.parseInt(arr[left][0])<pivot){
125                 left++;
126             }
127             while(Integer.parseInt(arr[right][0])>pivot){
128                 right--;
129             }
130             if(left<=right){
131                 for(int x=0;x<arr[0].length;x++){//loops through all fields in record to copy array
132                     buffer[x]=arr[right][x];
133                     arr[right][x]=arr[left][x];
134                     arr[left][x]=buffer[x];
135                 }
136                 left++;
137                 right--;
138             }
139         }
140         if(first<right){
141             quickSortInt(first, left-1,arr);
142         }
143         if(left<last){
144             quickSortInt(right+1, last,arr);
145         }        
146     }
