 27     public static UserInfo logInWithUserName(String userName, String password) throws Exception{
 28         Socket socket = new Socket(InetAddress.getByName(ChangeInternetSettings.getServerAddress()),ChangeInternetSettings.getPortNumber());//creates a socket to connect with the server
 29         ObjectInputStream in =new ObjectInputStream(socket.getInputStream());//gets the object input stream
 30         ObjectOutputStream out = new ObjectOutputStream(socket.getOutputStream());//gets the object output stream
 31         InputPackages pack = new InputPackages();pack.createLogInRequestForWithUserNameAndPassword(userName, password);//creates the package to send to the server
 32         out.writeObject(pack);//sends package to server
 33         return (UserInfo)in.readObject();//gets the UserInfo object from the server and returns it
 34     }
