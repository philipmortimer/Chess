 38     /**
 39      * Suggests a suitable game id. This will be either Game 1, Game 2 or Game 3.
 40      * @param ses The session info object used to get the appropriate username
 41      * @return The title
 42      */
 43     public static String suggestedGameTitle(SessionInfo ses){
 44         String title="Game ";int num=1;//stores the number to put at the end of the title
 45         String titlesTaken[]=new String[3];int noOfGames=0;//stores all game titles
 46         boolean isTaken=false;
 47         try{
 48             //gets the title of all current games
 49             FileReader read = new FileReader(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+ses.getUsername()+".txt");
 50             BufferedReader buffRead = new BufferedReader(read);
 51             String lineRead;
 52             while((lineRead=buffRead.readLine())!=null){
 53                 titlesTaken[noOfGames]=lineRead.split(UserInfo.SPLIT_ITEM)[0];
 54                 noOfGames++;
 55             }buffRead.close();read.close();
 56             //tries game1
 57             for(int i=0;i<noOfGames;i++){
 58                 if(titlesTaken[i].equals("Game1")){
 59                     isTaken=true;
 60                     break;
 61                 }
 62             }
 63             if(isTaken){
 64                 isTaken=false;
 65                 for(int i=0;i<noOfGames;i++){//tries Game2
 66                     if(titlesTaken[i].equals("Game2")){
 67                         isTaken=true;
 68                         break;
 69                     }
 70                 }
 71                 if(isTaken){
 72                     return "Game3";
 73                 }else{
 74                     return "Game2";
 75                 }
 76             }else{
 77                 return "Game1";
 78             }
 79         }catch(IOException e){
 80             JOptionPane.showMessageDialog(null,"An unexpected error occured "+e);
 81         }
 82         return title+String.valueOf(num);
 83     }
 84     /**
 85      * Checks to see if the game title chosen is valid (i.e. unique and not containing any illegal characters)
 86      * @param ses The session information
 87      * @param title The game title
 88      * @return True if valid, otherwise false
 89      */
 90     public static boolean isGameTitleValid(SessionInfo ses,String title){
 91         if(Validation.containsIllegalStrings(title)){//checks that the title does not contain illegal characters
 92             return false;
 93         }
 94         //searches to ensure that game title is unique
 95         try{
 96             FileReader read = new FileReader(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+ses.getUsername()+".txt");
 97             BufferedReader buffRead = new BufferedReader(read);
 98             String lineRead;
 99             while((lineRead=buffRead.readLine())!=null){
100                 if(lineRead.split(UserInfo.SPLIT_ITEM)[0].equals(title)){
101                     return false;
102                 }
103             }buffRead.close();read.close();
104         }catch(IOException e){
105             JOptionPane.showMessageDialog(null,"An unexpected error occured "+e);
106         }
107         return true;
108     }
109     /**
110      * Adds new offline game to text file
111      * @param gameTitle The game title
112      * @param opponentId The opponent id
113      * @param isUserWhite Whether the user plays as white
114      * @param secondsWhiteAndBlackHave The time control setting. This equals -100 if no time control
115      * @param sess The session info
116      */
117     public static void addNewGameToFile(String gameTitle,String opponentId, boolean isUserWhite,int secondsWhiteAndBlackHave,SessionInfo sess){
118         try{
119             //game id,opponentID, isUserWhite, boardState(s),time game has taken, time white has left, time black has left
120             //checks to see if file is empty
121             FileReader read = new FileReader(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+sess.getUsername()+".txt");
122             BufferedReader buffRead = new BufferedReader(read);
123             boolean isEmpty = buffRead.readLine()==null;buffRead.close();read.close();
124             //writes new record to text file
125             FileWriter write = new FileWriter(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+sess.getUsername()+".txt",true);
126             BufferedWriter buffWrite = new BufferedWriter(write);
127             if(isEmpty==false){
128                 buffWrite.newLine();
129             }
130             buffWrite.write(gameTitle+UserInfo.SPLIT_ITEM+opponentId+UserInfo.SPLIT_ITEM+String.valueOf(isUserWhite)+UserInfo.SPLIT_ITEM+OfflinePlay.NEW_BOARD+UserInfo.SPLIT_ITEM+"0"+UserInfo.SPLIT_ITEM+String.valueOf(secondsWhiteAndBlackHave)+UserInfo.SPLIT_ITEM+String.valueOf(secondsWhiteAndBlackHave));
131             buffWrite.flush();write.flush();buffWrite.close();write.close();
132         }catch(IOException e){
133             JOptionPane.showMessageDialog(null,"An unexpected error has occured "+e,"Error",JOptionPane.OK_OPTION);
134         }

458     private void createGameBtnActionPerformed(java.awt.event.ActionEvent evt) {                                              
459         // creates a new game after checking that game settings have actually been selected;
460         //gets selected difficulty setting
461         String computerDifficulty="";
462         if(difficultyOneBtn.isSelected()){
463             computerDifficulty=OfflinePlay.LEVEL_ONE_COMP;
464         }else if(difficultyTwoBtn.isSelected()){
465             computerDifficulty=OfflinePlay.LEVEL_TWO_COMP;
466         }else if(difficultyThreeBtn.isSelected()){
467             computerDifficulty=OfflinePlay.LEVEL_THREE_COMP;
468         }else if(difficultyFourBtn.isSelected()){
469             computerDifficulty=OfflinePlay.LEVEL_FOUR_COMP;
470         }else if(difficultyFiveBtn.isSelected()){
471             computerDifficulty=OfflinePlay.LEVEL_FIVE_COMP;
472         }else{
473             //error message if no difficulty setting selected
474             JOptionPane.showMessageDialog(this,"Please select a difficulty setting.","Error",JOptionPane.OK_OPTION);
475             return;
476         }
477         String gameTitle=gameTitleTxt.getText();//stores the game title
478         if(isGameTitleValid(session, gameTitle)==false||Validation.isPresent(gameTitle)==false){
479             JOptionPane.showMessageDialog(this,"Please choose a valid, unique game title.","Error",JOptionPane.OK_OPTION);
480             return;
481         }
482         boolean isUserWhite=false;//stores which colour the user will play as
483         if(whiteBtn.isSelected()){
484             isUserWhite=true;
485         }else if(blackBtn.isSelected()){
486             isUserWhite=false;
487         }else if(randomBtn.isSelected()){
488             //randomly assings user a colour
489             if(Math.random()>=0.5){
490                 isUserWhite=true;
491             }else{
492                 isUserWhite=false;
493             }
494         }else{
495             //error message if no colour chosen
496             JOptionPane.showMessageDialog(this,"Please select a colour to play as.","Error",JOptionPane.OK_OPTION);
497             return;           
498         }
499         int timeControl=-100;
500         if(unlimitedTimeBtn.isSelected()){
501             timeControl=-100;
502         }else if(fiveMinutesBtn.isSelected()){
503             timeControl=5*60;
504         }else if(tenMinsBtn.isSelected()){
505             timeControl=10*60;
506         }else{
507             JOptionPane.showMessageDialog(this,"Please select a time control setting.","Error",JOptionPane.OK_OPTION);
508             return;                   
509         }
510         addNewGameToFile(gameTitle,computerDifficulty,isUserWhite,timeControl,session);//adds game to file
511         //takes user to game
512         ChessBoardState[]state={new ChessBoardState(OfflinePlay.NEW_BOARD)};
513         ChessBoardOffline board = new ChessBoardOffline(session, computerDifficulty, new ChessGame(state,computerDifficulty.equals(OfflinePlay.LEVEL_TWO_COMP)), isUserWhite, 0, timeControl, timeControl, gameTitle);
514         board.setVisible(true);
515         this.dispose();
516     }
