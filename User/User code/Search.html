 14     /**
 15      * Performs a binary search on the one d array. Assumes it is sorted lexographically. 
 16      * @param DATA The data sorted lexographically
 17      * @param ITEM_TO_FIND the data item to be found
 18      * @return true is returned if the record is in the array, otherwise false is returned
 19      */
 20     public static boolean isRecordInArray(String[]DATA,String ITEM_TO_FIND){
 21         if(DATA.length==1){
 22             if(DATA[0].equals(ITEM_TO_FIND)){
 23                 return true;
 24             }else{
 25                 return false;
 26             }
 27         }
 28         if(DATA.length==0){
 29             return false;
 30         }
 31         String record="";
 32         int first = 0;
 33         int last = DATA.length-1;
 34         int mid;
 35         int comparison;
 36         String primaryKeyBuffer;
 37         do{
 38             mid = (first+last)/2;
 39             primaryKeyBuffer=DATA[mid];
 40             //if pk precedes itemToFind less than 0 
 41             comparison = primaryKeyBuffer.compareTo(ITEM_TO_FIND);
 42             if(comparison==0){
 43                 return true;
 44             }else if(comparison>0){
 45                 last = mid-1;
 46             }else{
 47                 first = mid + 1;
 48             }
 49         }while(last>=first);
 50         return false;
 51     }
 52     /**
 53      * Performs a binary search on the 2d array inputted. It searches for the itemToFind. It assumes this is a primary key at the nought element
 54      * of every row. It uses binary search and the java String method .compareTo(). Hence the data must be sorted by the primary key
 55      * with data of a small lexographic value at the top (i.e. sorted in ascending order).
 56      * @param DATA The data to be searched through.
 57      * @param ITEM_TO_FIND The item to find
 58      * @param WIDTH_OF_DATA The width of a record
 59      * @return the field containing the item to find. If no item is found, this will be a null array of width equal to the width of a record.
 60      */
 61     public static String[]searchForRecordWithPrimaryKeyAtNoughtElement(final String[][]DATA,final String ITEM_TO_FIND,final int WIDTH_OF_DATA){
 62         String record[]= new String[WIDTH_OF_DATA];
 63         if(DATA.length==0){
 64             return record;
 65         }
 66         int first = 0;
 67         int last = DATA.length-1;
 68         int mid;
 69         int comparison;
 70         String primaryKeyBuffer;
 71         do{
 72             mid = (first+last)/2;
 73             primaryKeyBuffer=DATA[mid][0];
 74             //if pk precedes itemToFind less than 0 
 75             comparison = primaryKeyBuffer.compareTo(ITEM_TO_FIND);
 76             if(comparison==0){
 77                 for(int element=0;element<record.length;element++){
 78                     record[element]=DATA[mid][element];
 79                 }
 80                 break;
 81             }else if(comparison>0){
 82                 last = mid-1;
 83             }else{
 84                 first = mid + 1;
 85             }
 86         }while(last>=first);
 87         return record;
 88     }
