 52     private void loadFormBasedOnIndex(){
 53         clearForm();//clears the form
 54         if(nonMemberInfo.length==0){//if there are no users who are not members, the form is updated to reflect this
 55             noUsersLbl.setVisible(true);
 56             addUserBtn.setEnabled(false);
 57             prevBtn.setEnabled(false);nextBtn.setEnabled(false);
 58             removeBtn.setEnabled(false);editBtn.setEnabled(false);
 59             return;
 60         }
 61         //enables or disables the buttons that allow navigation between users based on the user currently being viewed.
 62         prevBtn.setEnabled(true);nextBtn.setEnabled(true);
 63         addUserBtn.setEnabled(true);
 64         if(nonMemberIndex==0){
 65             prevBtn.setEnabled(false);
 66         }
 67         if(nonMemberIndex==nonMemberInfo.length-1){
 68             nextBtn.setEnabled(false);
 69         }
 70         //ensuers that the leader can't remove themselves
 71         removeBtn.setEnabled(true);
 72         if(nonMemberInfo[nonMemberIndex][0].equals(session.getUsername())){
 73             removeBtn.setEnabled(false);
 74         }
 75         //Username, password, first name, surname, postcode, addressline one, address line two, address line three, address line four, discord username, discord number, date of birth, phone number, isClubLeader, isClubMember, ELO, hasUserRequestedToJoinClub
 76         //sets the form fields with the appropriate values
 77         usernameTxt.setText(nonMemberInfo[nonMemberIndex][0]);
 78         emailAddressTxt.setText(emails[nonMemberIndex]);
 79         firstNameTxt.setText(nonMemberInfo[nonMemberIndex][2]);surnameTxt.setText(nonMemberInfo[nonMemberIndex][3]);
 80         postcodeTxt.setText(nonMemberInfo[nonMemberIndex][4]);addressLineOneTxt.setText(nonMemberInfo[nonMemberIndex][5]);
 81         addressLineTwoTxt.setText(nonMemberInfo[nonMemberIndex][6]);addressLineThreeTxt.setText(nonMemberInfo[nonMemberIndex][7]);
 82         addressLineFourTxt.setText(nonMemberInfo[nonMemberIndex][8]);discordUsernameTxt.setText(nonMemberInfo[nonMemberIndex][9]);
 83         discordNumberTxt.setText(nonMemberInfo[nonMemberIndex][10]);dateOfBirthTxt.setText(nonMemberInfo[nonMemberIndex][11]);
 84         phoneNumberTxt.setText(nonMemberInfo[nonMemberIndex][12]);eloTxt.setText(nonMemberInfo[nonMemberIndex][15]);
 85     }

775     /**
776      * Removes the user info of the user at the current index from the array and form. The index is then set to zero.
777      */
778     private void removeCurrentIndexUserFromArrayAndSetIndexToZero(){
779         //removes user info from array
780         String[][]newClubMemberInfo=new String[nonMemberInfo.length-1][UserInfo.WIDTH_OF_USER_DETAILS_FILE];
781         for(int i=0;i<nonMemberIndex;i++){
782             for(int x=0;x<UserInfo.WIDTH_OF_USER_DETAILS_FILE;x++){
783                 newClubMemberInfo[i][x]=nonMemberInfo[i][x];
784             }
785         }
786         for(int i=nonMemberIndex+1;i<nonMemberInfo.length;i++){
787             for(int x=0;x<UserInfo.WIDTH_OF_USER_DETAILS_FILE;x++){
788                 newClubMemberInfo[i-1][x]=nonMemberInfo[i][x];
789             }                    
790         }
791         //removes email from array
792         String[]newEmail=new String[emails.length-1];
793         for(int i=0;i<nonMemberIndex;i++){
794             newEmail[i]=emails[i];
795         }
796         for(int i=nonMemberIndex+1;i<emails.length;i++){
797             newEmail[i-1]=emails[i];
798         }
799         emails=newEmail;
800         nonMemberInfo=newClubMemberInfo;
801         nonMemberIndex=0;//sets index to zero
802     }
