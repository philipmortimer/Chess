 40     private void setFormBasedOnWhetherUserHasRequestedToJoinTheClub(){
 41         try{
 42             boolean hasRequested = ThreadClient.getHasUserRequestedToJoinChessClub(session.getUsername());//gets whether the user has requested to join the club
 43             //updates the button and label text based on whether the user has requested to join the club
 44             if(hasRequested){
 45                 requestToJoinClubBtn.setText("Withdraw request to join the West Cross Chess Club");
 46                 infoLbl.setText("Please feel free to contact the club or withdraw your request to join.");
 47             }else{
 48                 requestToJoinClubBtn.setText("Request to Join the West Cross Chess Club");
 49                 infoLbl.setText("Please feel free to contact the club or request to join.");
 50             }
 51         }catch(Exception e){
 52             error=true;//flags that an error has occurred
 53             JOptionPane.showMessageDialog(this,"An error occurred when communicating with the server. Error: "+e,"Error",JOptionPane.OK_OPTION);
 54             if(isFormInitialised){
 55                 Menu menu = new Menu(session);menu.setVisible(true);
 56                 this.dispose();
 57             }
 58         }
 59     }

 74             }

230     private void requestToJoinClubBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                     
231         // adds a request to join the West Cross Chess Club or withdraws it depending on the requests statis
232         try{
233             ThreadClient.changeRequestToJoinClubStatus(session.getUsername());//chanegs request status
234         }catch(Exception e){
235            JOptionPane.showMessageDialog(this,"An error occurred when attempting to communicate with the server. Error: "+e,"Error",JOptionPane.OK_OPTION);
236             error=true;
237             if(isFormInitialised){
