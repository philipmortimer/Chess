 49     public String getPortNumberString(){
 50         try{
 51             FileReader read = new FileReader(PORT_NUMBER_FILE_NAME);//reads the port number from the file
 52             BufferedReader buffRead = new BufferedReader(read);
 53             String ret = buffRead.readLine();
 54             buffRead.close();read.close();
 55             return ret;
 56         }catch(IOException e){
 57             JOptionPane.showMessageDialog(this, "An error occured with "+PORT_NUMBER_FILE_NAME, "Error", JOptionPane.OK_OPTION);
 58         }
 59         return "";
 60     }

 62      * Gets the port number
 63      * @return The port number as an integer. If an error occurs, an error message is displayed and -1 is returned.
 64      */
 65     public static int getPortNumber(){
 66         try{
 67             FileReader read = new FileReader(PORT_NUMBER_FILE_NAME);//reads the port number from the file
 68             BufferedReader buffRead = new BufferedReader(read);
 69             String portStr=buffRead.readLine();
 70             buffRead.close();read.close();
 71             try{
 72                 int port = Integer.parseInt(portStr);
 73                 return port;
 74             }catch(NumberFormatException ex){
 75                 JOptionPane.showMessageDialog(null, "The port number stored is not an integer. Please fix this by going into settings.", "Error", JOptionPane.OK_OPTION);
 76                 return -1;
 77             }
 78         }catch(IOException e){
 79             JOptionPane.showMessageDialog(null, "An error occured with "+PORT_NUMBER_FILE_NAME+". Error details: "+e, "Error", JOptionPane.OK_OPTION);
 80         }
 81         return -1;
 82     }

297 /**
298  * Saves the inputted changes after appropriate validation.
299  * @param evt The save button being pressed
300  */
301     private void saveChangesBtnActionPerformed(java.awt.event.ActionEvent evt) {                                               
302         try{
303             int port = Integer.parseInt(portNumberTxt.getText());
304             if(port<=0){//range check
305                 JOptionPane.showMessageDialog(this, "The port number must be a positive integer greater than zero.","Error",JOptionPane.OK_OPTION);
306                 return;
307             }
308         }catch(NumberFormatException ex){//type check
309             JOptionPane.showMessageDialog(this, "The port number must be a positive integer greater than zero.","Error",JOptionPane.OK_OPTION);
310             return;
311         }
312         try{
313             FileWriter writePort = new FileWriter(PORT_NUMBER_FILE_NAME,false);//writes port number and server address to their files
314             BufferedWriter buffWritePort = new BufferedWriter(writePort);
315             buffWritePort.write(portNumberTxt.getText());
316             buffWritePort.flush();writePort.flush();
317             buffWritePort.close();writePort.close();
318             FileWriter writeServ = new FileWriter(SERVER_ADDRESS_FILE_NAME,false);
319             BufferedWriter buffWriteServ = new BufferedWriter(writeServ);
320             buffWriteServ.write(serverAddressTxt.getText());
321             buffWriteServ.flush();writeServ.flush();
322             buffWriteServ.close();writeServ.close();
323         }catch(IOException e){
324             JOptionPane.showMessageDialog(this, "An unexpected error occured with either "+SERVER_ADDRESS_FILE_NAME+" or "+PORT_NUMBER_FILE_NAME+". Error Details "+e,"Error",JOptionPane.OK_OPTION);
325         }
326         isInEditMode=!isInEditMode;updateFormBasedOnWhetherInEditingMode();//changes editing mode and updates form accordingly
327     }
