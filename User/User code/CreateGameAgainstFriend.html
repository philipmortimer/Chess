223     private void createGameBtnActionPerformed(java.awt.event.ActionEvent evt) {                                              
224         //creates a new game after performing relevant validation
225         //gets time control to use
226         int timeLeft=-100;
227         if(fiveMinutesBtn.isSelected()){
228             timeLeft=5*60;
229         }else if(tenMinutesBtn.isSelected()){
230             timeLeft=600;
231         }else if(unlimitedBtn.isSelected()==false){
232             JOptionPane.showMessageDialog(this,"Please select a time control.","Error",JOptionPane.OK_OPTION);
233             return;
234         }
235         String userName="";//stores the username of the opponent
236         if(useCodeOrUsernameBtn.getText().equals("Create Code to Share Instead")){
237             userName=friendUserNameTxt.getText();
238             if(!Validation.isPresent(userName) || userName.equals(session.getUsername())|| userName.equals(UserInfo.GUEST_ID)){//ensures that the username can't be empty or be that of the current user
239                 JOptionPane.showMessageDialog(this,"Please input a valid username.","Error",JOptionPane.OK_OPTION);
240                 return;
241             }
242             try{
243                 if(ThreadClient.isUserNameAlreadyTaken(userName)==false){//checks to see if username is taken
244                     JOptionPane.showMessageDialog(this,"User not found. Please enter  a valid username.","Error",JOptionPane.OK_OPTION);
245                     return;
246                 }
247                 if(ThreadClient.getOnlineGameData(userName)[2][0].equals("")==false){//checks to see if friend is already playing three games
248                     JOptionPane.showMessageDialog(this,"User selected is currently playing three games. Users can not play more than three games.","Error",JOptionPane.OK_OPTION);
249                     return;
250                 }
251             }catch(Exception e){
252                 JOptionPane.showMessageDialog(this, "There has been an error when communicating with the server.","Error",JOptionPane.OK_OPTION);
253                 return;
254             }
255         }
256         try{
257             String code=ThreadClient.createGameAgainstFriend(session.getUsername(), userName, timeLeft);//creates the game
258             if(code.equals("-1")){
259                 JOptionPane.showMessageDialog(this, "There has been an error when communicating with the server.","Error",JOptionPane.OK_OPTION);
260             }else if(userName.equals("")){
261                 JOptionPane.showMessageDialog(this,"Game created. Game code "+code,"Game Created",JOptionPane.OK_OPTION);
262             }else{
263                 JOptionPane.showMessageDialog(this,"Game created.","Game Created",JOptionPane.OK_OPTION);
264             }
265         }catch(Exception e){
266             JOptionPane.showMessageDialog(this, "There has been an error when communicating with the server.","Error",JOptionPane.OK_OPTION);           
267         }
268         //takes user back to online menu
269         OnlineMenu online = new OnlineMenu(session);
270         online.setVisible(true);
271         this.dispose();
272         
273     }
