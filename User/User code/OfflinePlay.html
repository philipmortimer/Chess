 59     private void updateTable(){
 60         try{
 61             File file = new File(OFFLINE_GAMES_FOLDER+File.separator+sess.getUsername()+".txt");
 62             file.createNewFile();//this creates a new text file if one does not already exist
 63         }catch(IOException ex){
 64             JOptionPane.showMessageDialog(this,"An unexpected error occured "+ex,"Error",JOptionPane.OK_OPTION);
 65         }
 66         try{
 67             //gets game data from text file and loads it global 2d array
 68             FileReader read = new FileReader(OFFLINE_GAMES_FOLDER+File.separator+sess.getUsername()+".txt");
 69             BufferedReader buffRead = new BufferedReader(read);
 70             String lineRead;String lineArr[];
 71             gameData=new String[3][7];
 72             int in=0;
 73             //loads game data into array
 74             while((lineRead=buffRead.readLine())!=null){
 75                 lineArr = lineRead.split(UserInfo.SPLIT_ITEM);
 76                 for(int i=0;i<lineArr.length;i++){
 77                     gameData[in][i]=lineArr[i];
 78                 }
 79                 in++;
 80             }
 81             buffRead.close();read.close();
 82         }catch(IOException ex){
 83             JOptionPane.showMessageDialog(this,"An unexpected error occured "+ex,"Error",JOptionPane.OK_OPTION);
 84         }
 85         //updates table contents to reflect gameData array
 86         String[]games;//stores all games that have been played so far
 87         for(int y=0;y<gameData.length;y++){
 88             if(gameData[y][0]==null){
 89                 offlineGamesTbl.setValueAt("", y, 0);//sets game title as empty
 90                 offlineGamesTbl.setValueAt("",y,1);//sets opponent empty
 91                 offlineGamesTbl.setValueAt("",y,2);//sets move number empty
 92             }else{
 93                 offlineGamesTbl.setValueAt(gameData[y][0], y, 0);//sets game title
 94                 offlineGamesTbl.setValueAt(gameData[y][1],y,1);//sets opponent id
 95                 games=gameData[y][3].split(SPLIT_FEN_STRING);
 96                 offlineGamesTbl.setValueAt(games[games.length-1].split(" ")[5],y,2);//sets move number
 97             }
 98         }
 99     }

305     private void playGameBtnActionPerformed(java.awt.event.ActionEvent evt) {                                            
306         //takes user to selected game, if it is a valid game
307         if(offlineGamesTbl.getSelectedRow()==-1 || gameData[offlineGamesTbl.getSelectedRow()][0]==null){//checks to make sure a game is actually selected
308             JOptionPane.showMessageDialog(this, "Please select a valid game to play.","No Game Selected",JOptionPane.OK_OPTION);
309             return;
310         }
311         String fen[]=gameData[offlineGamesTbl.getSelectedRow()][3].split(SPLIT_FEN_STRING);//stores all fen strings for games so far
312         ChessBoardState states[]=new ChessBoardState[fen.length];//stores all boards played so far
313         for(int i=0;i<states.length;i++){//loads all states
314             states[i]=new ChessBoardState(fen[i]);
315         }
316         //creates new chess board
317         ChessBoardOffline board = new ChessBoardOffline(sess, gameData[offlineGamesTbl.getSelectedRow()][1], new ChessGame(states,gameData[offlineGamesTbl.getSelectedRow()][1].equals(LEVEL_TWO_COMP)), gameData[offlineGamesTbl.getSelectedRow()][2].equals("true"), Integer.parseInt(gameData[offlineGamesTbl.getSelectedRow()][4]),Integer.parseInt(gameData[offlineGamesTbl.getSelectedRow()][5]),Integer.parseInt(gameData[offlineGamesTbl.getSelectedRow()][6]),gameData[offlineGamesTbl.getSelectedRow()][0]);
318         board.setVisible(true);this.dispose();
319 
320     }

322     private void deleteGameBtnActionPerformed(java.awt.event.ActionEvent evt) {                                              
323         //deletes game stored if it is a valid game to delete
324         if(offlineGamesTbl.getSelectedRow()==-1 || gameData[offlineGamesTbl.getSelectedRow()][0]==null){//checks to make sure a game is actually selected
325             JOptionPane.showMessageDialog(this, "Please select a valid game to delete.","No Game Selected",JOptionPane.OK_OPTION);
326             return;
327         }
328         //seeks user confirmation before deleting game
329         if(JOptionPane.showConfirmDialog(this, "Are you sure you would like to delete this game?", "Delete?", JOptionPane.YES_NO_CANCEL_OPTION)!=JOptionPane.YES_OPTION){
330             return;
331         }
332         //deletes game from text file and array and uptables table accordingly
333         try{
334             //writes old data without the game deleted to the offline game file
335             FileWriter write = new FileWriter(OFFLINE_GAMES_FOLDER+File.separator+sess.getUsername()+".txt",false);
336             BufferedWriter buffWrite = new BufferedWriter(write);
337             boolean isFirstWrite=true;
338             String line;//stores line of text to write
339             //writes new data to file
340             for(int y =0;y<gameData.length;y++){
341                 if(y!=offlineGamesTbl.getSelectedRow()){
342                     if(gameData[y][0]!=null){
343                         line=gameData[y][0];
344                         for(int i=1;i<gameData[0].length;i++){//creates text to write to file
345                             line =line+ UserInfo.SPLIT_ITEM+gameData[y][i];
346                         }
347                         if(isFirstWrite==false){
348                             buffWrite.newLine();
349                         }else{
350                             isFirstWrite=false;
351                         }
352                         buffWrite.write(line);
353                     }
354                 }
355             }
356             buffWrite.flush();write.flush();buffWrite.close();write.close();//closes and flushes writers
357         }catch(IOException ex){
358             JOptionPane.showMessageDialog(this,"An unexpected error occured "+ex,"Error",JOptionPane.OK_OPTION);
359         }
360         updateTable();//updates table and gameData array
361     }
