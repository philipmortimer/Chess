229     public void createAccountPacket(String []userRecord,String userName, String emailAddress){
230         packageToSend=new String[userRecord.length+3];
231         packageToSend[0]=CREATE_ACCOUNT;
232         for(int i=0;i<userRecord.length;i++){

263     /**

265      * @return A copy of the package
266      */
267     public InputPackages getCopyOfInputPackage(){
268         String oldArray[]=this.getPackageToSend();
269         String newArray[]=new String[oldArray.length];//copies previous array
270         for(int i=0;i<newArray.length;i++){
271             newArray[i]=oldArray[i];
272         }
273         return new InputPackages(newArray);
274     }
