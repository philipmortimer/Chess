 16     /**
 17      * Checks to see whether input string is empty or not. Returns false if length of String is 0 or a NullPointerException occurs (e.g. for newly intialised String arrays java).
 18      * Otherwise true is returned. Note: this presence check views a String with a (or any) single space(s) in it as 
 19      * a non-empty input and returns true for it.
 20      * @param stringToCheck the string to be checked
 21      * @return returns false if the String is empty (i.e. has a length of 0, using the .length() command or triggers a NullPointerException caused by String arrays that have been initialised but have no value assigned to them) and true if the string is not empty
 22      */
 23     public static boolean isPresent(String stringToCheck){//presence check
 24         try{
 25             if(stringToCheck.length()==0){//checks to see if number of characters are zero e.g.:""
 26                 return false;
 27             }else{
 28                 return true;
 29             }
 30         }catch(java.lang.NullPointerException e){//nullpointer exception triggered for newly initialised elements in a string array
 31             return false;
 32         }
 33     }

 75     /**
 76      * Performs a linear search through all acceptable inputs comparing it to the String input. If a match is found, true is returned, else false is returned.
 77      * @param acceptableInputs A String array containing all acceptable inputs
 78      * @param stringToCheck The String to check against the acceptable inputs
 79      * @return True is returned if stringToCheck matches any item in the array, otherwise false is returned.
 80      */
 81     public static boolean lookUpCheck(String[]acceptableInputs,String stringToCheck){//lookup check
 82         for(int indexInAcceptableInputs=0;indexInAcceptableInputs<acceptableInputs.length;indexInAcceptableInputs++){
 83             if(stringToCheck.equals(acceptableInputs[indexInAcceptableInputs])){
 84                 return true;
 85             }
 86         }
 87         return false;
 88     }

139     /**
140      * Checks to see if the postcode is valid.
141      * @param postCode The postcode as a string
142      * @return True is returned if the postcode is between 6 and 8 characters, contains exactly one space and all characters other than a space character are capitalised.
143      */
144     public static boolean isValidPostcode(String postCode){
145         if(postCode.length()>8 || postCode.length()<6){//length checks
146             return false;
147         }
148         int noOfSpaces=0;//counts number of spaces
149         for(int c=0;c<postCode.length();c++){
150             if(postCode.charAt(c)==' '){
151                 if(noOfSpaces==1){
152                     return false;
153                 }
154                 noOfSpaces++;
155             }else{
156                 if(!Character.isUpperCase(postCode.charAt(c)) && Character.isDigit(postCode.charAt(c))==false){
157                     return false;
158                 }    
159             }
160         }
161         if(noOfSpaces==0){
162             return false;
163         }
164         return true;
165     }

172         for(int c=0;c<phoneNumber.length();c++){
173             if(phoneNumber.charAt(c)!=' '&& phoneNumber.charAt(c)!='+'&&!Character.isDigit(phoneNumber.charAt(c))){
174                 return false;
175             }
176         }
177         return true;
