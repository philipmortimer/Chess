513     private void approveBtnActionPerformed(java.awt.event.ActionEvent evt) {                                           
514         // approves the request to join the club
515         try{
516             ThreadClient.withdrawRequestToJoinClub(usernameTxt.getText());
517             ThreadClient.addUserToClubAsMember(usernameTxt.getText());
518         }catch(Exception e){
519             JOptionPane.showMessageDialog(this,"An error occurred when communicating with the server: "+e,"Error",JOptionPane.OK_OPTION);
520             Menu menu = new Menu(session);menu.setVisible(true);this.dispose();
521         }
522         //updates the form following the acceptance of the request
523         clearForm();
524         currentReqeustIndex++;
525         checkIfThereAreMoreRequestsAndUpdateComponents();
526         loadRequestToFormIfAppropraite();
527     }
