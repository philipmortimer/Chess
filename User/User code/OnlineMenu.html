 74     private void updateTable(){
 75         try{
 76             String[]games;
 77             //sets table to empty first
 78             for(int y=0;y<3;y++){
 79                 for(int x=0;x<3;x++){
 80                     onlineGamesTbl.setValueAt("",y,x);
 81                 }
 82             }
 83             //stores whether a given row should be highlighted based on whether a game actually exists in that row of the table
 84             boolean isRowOneHighlighted=true;
 85             boolean isRowTwoHighlighted=true;
 86             boolean isRowThreeHighlighted=true;
 87             gameData=ThreadClient.getOnlineGameData(session.getUsername());//gets game data from server
 88             //updates table based on game data array
 89             for(int y=0;y<gameData.length;y++){
 90                 onlineGamesTbl.setValueAt(gameData[y][0],y,0);//sets game code
 91                 //sets the username of the opponent
 92                 if(session.getUsername().equals(gameData[y][1])){
 93                     onlineGamesTbl.setValueAt(gameData[y][2],y,1);
 94                 }else{
 95                     onlineGamesTbl.setValueAt(gameData[y][1],y,1);
 96                 }
 97                 games=gameData[y][3].split(OfflinePlay.SPLIT_FEN_STRING);
 98                 if(String.valueOf(onlineGamesTbl.getValueAt(y, 1)).equals("")){//if the game has not yet been created, the field is set as empty and the row is rendered as a different colour
 99                     switch (y) {
100                         case 0:
101                             isRowOneHighlighted=false;
102                             break;
103                         case 1:
104                             isRowTwoHighlighted=false;
105                             break;
106                         default:
107                             isRowThreeHighlighted=false;
108                             break;
109                     }
110                     onlineGamesTbl.setValueAt("",y,2);
111                 }else{
112                     onlineGamesTbl.setValueAt(games[games.length-1].split(" ")[5],y,2);//gets the move number
113                 }
114             }
115             renderer.updateRowColour(isRowOneHighlighted, isRowTwoHighlighted, isRowThreeHighlighted);//updates highlighter
116         }catch(Exception e){
117             JOptionPane.showMessageDialog(this,"An error occured when retriving game data from the server. Please check your network settings. Error Details: "+e.getMessage(),"Error",JOptionPane.OK_OPTION);
118             Menu menu = new Menu(session);
119             menu.setVisible(true);
120             this.dispose();return;//displays error message and takes user back to menu
121         }
122     }

333     private void deleteBtnActionPerformed(java.awt.event.ActionEvent evt) {                                          
334         //attempts to delete a game that can be deleted.
335         //only games where the opponent has net yet joined can be deleted
336         if(onlineGamesTbl.getSelectedRow()>=0 && onlineGamesTbl.getSelectedRow()<=2){
337             if(onlineGamesTbl.getValueAt(onlineGamesTbl.getSelectedRow(),1).equals("")==false){

342             int confirm=JOptionPane.showConfirmDialog(this,"Are you sure you would like to delete this game?","Delete?",JOptionPane.YES_NO_CANCEL_OPTION);
343             if(confirm!=JOptionPane.YES_OPTION){
344                 return;
345             }
346             try{
347                 ThreadClient.deleteOnlineGame(Integer.parseInt(String.valueOf(onlineGamesTbl.getValueAt(onlineGamesTbl.getSelectedRow(),0))), 0.5);//deletes game
348                 updateTable();//updates the tab;e
349             }catch(Exception  e){
350                 JOptionPane.showMessageDialog(this,"Error when communicating with the server. Please check your network settings. Error details: "+e.getMessage(),"Error",JOptionPane.OK_OPTION);

352                 return;
353             }

355             JOptionPane.showMessageDialog(this, "Please select a valid game to delete","Error",JOptionPane.OK_OPTION);

371     private void playGameAgainstFriendEnterCodeBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                                  
372         // allows user to input game code and play against friend
373         if(onlineGamesTbl.getValueAt(2,0).equals("")==false){//ensures that user does not join more than three games
374             JOptionPane.showMessageDialog(this, "You can only have three games at any one time.","Error",JOptionPane.OK_OPTION);
375             return;
376         }

378         if(code==null){//if cancel is pressed, nothing happens
379             return;
380         }
381         try{
382             int codeInt=Integer.parseInt(code);//tries converting code to an integer
383             if(codeInt<0){
384                 //type and range check to ensure that the game code is possibly valid
385                 JOptionPane.showMessageDialog(this,"The game code must be a valid integer greater than 0","Invalid code",JOptionPane.OK_OPTION);
386                 return;
387             }
388             try{
389                 boolean succ = ThreadClient.joinGameUsingFriendCode(codeInt, session.getUsername());//attempts to join the game
390                 if(succ==false){
391                     JOptionPane.showMessageDialog(this,"Unable to join game. The code entered is invalid or can't be used to join a game","Error",JOptionPane.OK_OPTION);
392                     updateTable();
393                     return;
394                 }
395                 updateTable();//updates the table
396             }catch(Exception e){
397                  JOptionPane.showMessageDialog(this,"Error when communicating with the server. Please check your network settings. Error details: "+e.getMessage(),"Error",JOptionPane.OK_OPTION);
398                  updateTable();
399                  return;
400             }
401         }catch(NumberFormatException e){
402             JOptionPane.showMessageDialog(this,"The game code must be a valid integer greater than 0","Invalid code",JOptionPane.OK_OPTION);
403             return;
404         }

407     private void playBtnActionPerformed(java.awt.event.ActionEvent evt) {                                        
408         // takes the user to the selected game
409         if(onlineGamesTbl.getSelectedRow()<0 || onlineGamesTbl.getSelectedRow()>2){//ensures that a valid game is selected
410             JOptionPane.showMessageDialog(this, "Invalid game selected.","Invalid Game",JOptionPane.OK_OPTION);
411             return;
412         }
413         //checks to ensure game selected is playable
414         if(onlineGamesTbl.getValueAt(onlineGamesTbl.getSelectedRow(), 0).equals("") || onlineGamesTbl.getValueAt(onlineGamesTbl.getSelectedRow(), 1).equals("")){
415             JOptionPane.showMessageDialog(this, "Invalid game selected.","Invalid Game",JOptionPane.OK_OPTION);
416             return;            
417         }
418         //gets the most recent version of the data from the server
419         String []data=null;
420         try{
421             data=ThreadClient.getGameData(Integer.parseInt(String.valueOf(onlineGamesTbl.getValueAt(onlineGamesTbl.getSelectedRow(),0))));//gets updated version of game record before joining
422         }catch(Exception e){
423             JOptionPane.showMessageDialog(this, "Error when communicating with the server. Error details: "+e,"Invalid Game",JOptionPane.OK_OPTION);
424             return;
425         }
426         //checks to see if game is already over - if it is, an appropriate message is displayed and the game is deleted
427         boolean isUserPlayerOne=data[1].equals(session.getUsername());
428         if(isUserPlayerOne && data[8].equals("false") && data[9].equals("true")){
429             if(data[11].equals("1")){
430                 JOptionPane.showMessageDialog(this,"You won this game.","Win",JOptionPane.OK_OPTION);
431             }else if(data[11].equals("0.5")){
432                 JOptionPane.showMessageDialog(this,"This game was a draw.","Draw",JOptionPane.OK_OPTION);
433             }else{
434                 JOptionPane.showMessageDialog(this,"You lost this game.","Loss",JOptionPane.OK_OPTION);
435             }
436             data[8]="true";
437             try{
438                 ThreadClient.updateOnlineGameRecord(data);
439             }catch(Exception e){
440                 JOptionPane.showMessageDialog(this, "Error when  communicating with the server. Error details: "+e,"Error",JOptionPane.OK_OPTION);
441                 updateTable();
442                 return;
443             }
444             updateTable();
445             return;           
446         }else if(!isUserPlayerOne && data[9].equals("false") && data[8].equals("true")){
447             if(data[11].equals("1")){
448                 JOptionPane.showMessageDialog(this,"You lost this game.","Loss",JOptionPane.OK_OPTION);
449             }else if(data[11].equals("0.5")){
450                 JOptionPane.showMessageDialog(this,"This game was a draw.","Draw",JOptionPane.OK_OPTION);
451             }else{
452                 JOptionPane.showMessageDialog(this,"You won this game.","Win",JOptionPane.OK_OPTION);
453             }
454             data[9]="true";
455             try{
456                 ThreadClient.updateOnlineGameRecord(data);
457             }catch(Exception e){
458                 JOptionPane.showMessageDialog(this, "Error when  communicating with the server. Error details: "+e,"Error",JOptionPane.OK_OPTION);
459                 updateTable();
460                 return;
461             }
462             updateTable();
463             return;
464         }
465         //games is playable, hence user is taken to it
466         //creates the chess game object
467         String fenData[]=data[3].split(OfflinePlay.SPLIT_FEN_STRING);
468         ChessBoardState[]statesSoFar=new ChessBoardState[fenData.length];
469         for(int board=0;board<fenData.length;board++){
470             statesSoFar[board]=new ChessBoardState(fenData[board]);
471         }
472         //takes the user to play the game
473         ChessGame game = new ChessGame(statesSoFar, false);
474          ChessBoardOnline   onlineGame = new ChessBoardOnline(session,game,data);onlineGame.setVisible(true);this.dispose();
