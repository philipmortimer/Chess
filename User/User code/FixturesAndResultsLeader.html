257      * Updates the fixtures array game results based on the JTable. This only updates the local array.
258      */
259     private void updateFixturesArray(){
260         //loops through all results in the current round and updates the fixture/results array.
261         //this essentially results in any changes to the game states being saved to the array.
262         String result;
263         for(int match=0;match<fixturesAndResults[roundIndex][0].length;match++){
264             result=fixturesAndResultsTbl.getValueAt(match, 2).toString();
265             //converts result to computer representation of game state
266             switch (result) {
267                 case BLACK_WIN:
268                     fixturesAndResults[roundIndex][2][match]=Fixture.GAME_BLACK_WIN;
269                     break;
270                 case WHITE_WIN:
271                     fixturesAndResults[roundIndex][2][match]=Fixture.GAME_WHITE_WIN;
272                     break;
273                 case GAME_DRAWN:
274                     fixturesAndResults[roundIndex][2][match]=Fixture.GAME_DRAW;
275                     break;
276                 case GAME_NOT_YET_PLAYED:
277                     fixturesAndResults[roundIndex][2][match]=Fixture.GAME_NOT_YET_PLAYED;
278                     break;
279                 default:
280                     //if game is a bye
281                     fixturesAndResults[roundIndex][2][match]=Fixture.BYE_GAME_STATE;
282                     break;
283             }
284         }
285     }

288      * @throws Exception An exception may occur when attempting to retrieve information from the server

290     private void updateVariablesServer() throws Exception{
291         fixturesAndResults= ThreadClient.getFixturesAndResults();//gets fixtures/results from the server
292         haveFixturesBeenGenerated=fixturesAndResults.length!=0;//determines whether the fixtuers have actually been generated. A fixtures array of length 0 indicates that the fxitures have not been generated.
293         roundIndex=0;//sets the round index to 0
294     }
295     /**
296      * Updates the table based on the data stored in the object
297      */
298     private void updateTable(){
299         if(haveFixturesBeenGenerated){
300             //displays the fixtyres/results of the current round in the table
301             roundLbl.setText("Round "+String.valueOf(roundIndex+1));//displays which round number is being viewed
302             //enables/disables the previous and next round buttons based on whether the previous/next round exists
303             prevRoundBtn.setVisible(true);
304             prevRoundBtn.setEnabled(roundIndex!=0);
305             nextRoundBtn.setEnabled(roundIndex!=(fixturesAndResults.length-1));
306             nextRoundBtn.setVisible(true);
307             //displays the table
308             jScrollPane1.setEnabled(true);jScrollPane1.setVisible(true);
309             fixturesAndResultsTbl.setEnabled(true);fixturesAndResultsTbl.setVisible(true);
310             //uses the fixtures and results array to generate the table data
311             String[][]tableData=new String[fixturesAndResults[0][0].length][COLUMN_NAMES.length];
312             //in the form: white, black, result
313             for(int player=0;player<tableData.length;player++){
314                 tableData[player][0]=fixturesAndResults[roundIndex][0][player];
315                 if(tableData[player][0].equals(Fixture.BY_NAME)){//bye player names are replaced with an empty string
316                     tableData[player][0]="";
317                 }
318                 tableData[player][1]=fixturesAndResults[roundIndex][1][player];
319                 if(tableData[player][1].equals(Fixture.BY_NAME)){
320                     tableData[player][1]="";
321                 }
322                 //converts the game state from a computer form to a form that is understandable by humans
323                 switch (fixturesAndResults[roundIndex][2][player]) {
324                     case Fixture.BYE_GAME_STATE:
325                         tableData[player][2]=BYE_GAME;
326                         break;
327                     case Fixture.GAME_DRAW:
328                         tableData[player][2]=GAME_DRAWN;
329                         break;
330                     case Fixture.GAME_WHITE_WIN:
331                         tableData[player][2]=WHITE_WIN;
332                         break;
333                     case Fixture.GAME_BLACK_WIN:
334                         tableData[player][2]=BLACK_WIN;
335                         break;
336                     default:
337                         //game has not yet been played
338                         tableData[player][2]=GAME_NOT_YET_PLAYED;
339                         break;
340                 }
341             }
342             JComboBox <String>game = new JComboBox<String>(GAME_STATE_OPTIONS);//creates a new ComboBox that allows the game results to be edited by the leader
343             DefaultTableModel model = new DefaultTableModel(tableData, COLUMN_NAMES){//creates a new table model using the data generated
344                 /**
345                  * Determines whether the cell is editable. Only game results can be edited. Only game results that contain two human players can be edited.
346                  * @param row The row number of the cell
347                  * @param column The column number of the cell
348                  * @return Whether the cell is editable
349                  */
350                 @Override
351                 public boolean isCellEditable(int row, int column){
352                     //a game result can only be altered if the game is against two human players
353                     return !(column==2 && (tableData[row][1].equals("") || tableData[row][0].equals("")));
354                 }
355             };
356             fixturesAndResultsTbl.setModel(model);//sets the table to the generated model
357             fixturesAndResultsTbl.getColumnModel().getColumn(2).setCellEditor(new DefaultCellEditor(game));//sets it so that the game results are edited using a JComboBox. This avoids the need to validate inputs as only valid game states can be selected.
358             return;
359         }else{
360             //updates the form to display to the user that the fixtures have not yet been generated
361             roundLbl.setText("The fixtures have not yet been generated");
362             saveBtn.setVisible(false);saveBtn.setEnabled(false);
363             prevRoundBtn.setEnabled(false);prevRoundBtn.setVisible(false);
364             nextRoundBtn.setEnabled(false);nextRoundBtn.setVisible(false);
365             fixturesAndResultsTbl.setEnabled(false);fixturesAndResultsTbl.setVisible(false);
366             jScrollPane1.setEnabled(false);jScrollPane1.setVisible(false);
367             return;
368         }
