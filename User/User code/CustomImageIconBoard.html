 8 import java.awt.Component;
 9 import java.awt.Graphics;
10 import java.awt.Graphics2D;
11 import java.awt.Image;
12 import java.awt.RenderingHints;
13 import java.awt.image.BufferedImage;
14 import javax.swing.ImageIcon;

16 /**
17  * A custom image icon class that ensures the image icon of chess pieces of high quality
18  * @author mortimer
19  */
20 public class CustomImageIconBoard extends ImageIcon {
21     /**
22      * Creates an ImageIcon using a buffered image. This image is then rendered using the highest quality rendering settings
23      * @param f The 56 by 56 pixel buffered image.
24      */
25     public CustomImageIconBoard(BufferedImage f) {
26       super(f);
27       //sets a high render quality of the ImageIcon
28       BufferedImage i= new BufferedImage(56, 56, BufferedImage.TYPE_INT_ARGB);
29       Graphics2D g2d = (Graphics2D)i.getGraphics();
30                 g2d.setRenderingHint(RenderingHints.KEY_ALPHA_INTERPOLATION, RenderingHints.VALUE_ALPHA_INTERPOLATION_QUALITY);
31                 g2d.setRenderingHint(RenderingHints.KEY_COLOR_RENDERING, RenderingHints.VALUE_COLOR_RENDER_QUALITY);
32                 g2d.setRenderingHint(RenderingHints.KEY_DITHERING, RenderingHints.VALUE_DITHER_ENABLE);
33                 g2d.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS, RenderingHints.VALUE_FRACTIONALMETRICS_ON);
34                 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
35                 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
36                 g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
37                 
38       g2d.drawImage(getImage(), 0, 0, 56, 56, null);//draws the fifty six by fifty six ImageIcon
39       setImage(i);
40     }
41     /**
42      * Creates a custom image icon that renders an image at the highes possible quality. 
43      * @param f the 56 by 56 pixel image
44      */
45     public CustomImageIconBoard(Image f) {
46       super(f);
47       //sets a high render quality of the ImageIcon
48       BufferedImage i= new BufferedImage(56, 56, BufferedImage.TYPE_INT_ARGB);
49       Graphics2D g2d = (Graphics2D)i.getGraphics();
50                 g2d.setRenderingHint(RenderingHints.KEY_ALPHA_INTERPOLATION, RenderingHints.VALUE_ALPHA_INTERPOLATION_QUALITY);
51                 g2d.setRenderingHint(RenderingHints.KEY_COLOR_RENDERING, RenderingHints.VALUE_COLOR_RENDER_QUALITY);
52                 g2d.setRenderingHint(RenderingHints.KEY_DITHERING, RenderingHints.VALUE_DITHER_ENABLE);
53                 g2d.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS, RenderingHints.VALUE_FRACTIONALMETRICS_ON);
54                 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
55                 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
56                 g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
57                 
58       g2d.drawImage(getImage(), 0, 0, 56, 56, null);
59       setImage(i);
60     }
61     /**
62      * Gets the icon's height.
63      * @return The icon's height (this will be 56)
64      */
65     @Override
66     public int getIconHeight() {
67       return 56;
68     }
69     /**
70      * Gets the icon's width.
71      * @return The icon's width (this will be 56)
72      */
73     @Override
74     public int getIconWidth() {
75       return 56;
76     }
77     /**
78      * Paints the icon
79      * @param c The component
80      * @param g The graphics
81      * @param x The x value
82      * @param y The y value
83      */
84     @Override
85     public void paintIcon(Component c, Graphics g, int x, int y) {
86       g.drawImage(getImage(), x, y, c);
87     }   
88 }
