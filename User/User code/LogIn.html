202     private void showOrHidePasswordBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                      
203         //hides or makes the password visible
204         showPassword=!showPassword;
205         if(showPassword){
206             //makes the password visible as plain text
207             showOrHidePasswordBtn.setText("Hide Password");
208             char c=0;
209             passwordTxt.setEchoChar(c);
210         }else{
211             //masks the password
212             showOrHidePasswordBtn.setText("Show Password");
213             passwordTxt.setEchoChar(DEFAULT_ECHO_CHAR);
214         }
215     }

236     private void logInBtnActionPerformed(java.awt.event.ActionEvent evt) {                                         
237         //logs user in via server
238         //validates input somewhat before querying server
239         if(usernameOrEmailAddressTxt.getText().equals(UserInfo.GUEST_ID)|| usernameOrEmailAddressTxt.getText().equals(UserInfo.GUEST_ID_EMAIL)){//banned inputs reserved for guest ids
240             JOptionPane.showMessageDialog(this,"You can't log in with this ID.","Error",JOptionPane.OK_OPTION);
241             return;
242         }
243         if(usernameOrEmailAddressTxt.getText().contains("@")){//for email address log in
244             try {
245                 //checks to see if email address exists
246                 if(ThreadClient.isEmailAddressAlreadyTaken(usernameOrEmailAddressTxt.getText())==false){
247                     JOptionPane.showMessageDialog(this, "The email address was not found.","Error",JOptionPane.OK_OPTION);
248                     return;                     
249                 }
250                 //attempts to log user in
251                 UserInfo user = ThreadClient.logInWithEmailAddress(usernameOrEmailAddressTxt.getText(), new String(passwordTxt.getPassword()));
252                 if(user.getIsPasswordCorrect()==false){//if password does not match email
253                     JOptionPane.showMessageDialog(this, "The password is incorrect.","Error",JOptionPane.OK_OPTION);
254                     return;
255                 }
256                 //takes user to menu if log in was succesful
257                 Menu menu = new Menu(new SessionInfo(user.getUserName(), user.getIsClubMember(), user.getIsClubLeader()));menu.setVisible(true);this.dispose();return;//takes user to menu with appropriate session information
258             } catch (Exception ex) {
259                 JOptionPane.showMessageDialog(this, "A network error has occured. This may be due to your network settings or the server. "+ex,"Error",JOptionPane.OK_OPTION);   
260                 return;
261             }
262         }else{//for username
263             try{
264                 //attempts to log user in with username
265                 UserInfo user = ThreadClient.logInWithUserName(usernameOrEmailAddressTxt.getText(), new String(passwordTxt.getPassword()));//gets user info from server
266                 if(user.getUserFound()==false){//displays appropraite message if the username was not found
267                     JOptionPane.showMessageDialog(this, "The username was not found.","Error",JOptionPane.OK_OPTION);
268                     return;
269                 }
270                 if(user.getIsPasswordCorrect()==false){//displays appropraite message if the password entered was wrong
271                     JOptionPane.showMessageDialog(this, "The password is incorrect.","Error",JOptionPane.OK_OPTION);
272                     return;
273                 }
274                 Menu menu = new Menu(new SessionInfo(user.getUserName(), user.getIsClubMember(), user.getIsClubLeader()));menu.setVisible(true);this.dispose();return;//takes user to menu with appropriate session information
275                 //takes user to menu having created session info object to store relevant data
276             }catch(Exception e){
277                 JOptionPane.showMessageDialog(this, "A network error has occured. This may be due to your network settings or the server. "+e,"Error",JOptionPane.OK_OPTION);   
278                 return;
279             }
280         }
281     }
