 44     private void addNewGameToFile(String opponentId, boolean isUserWhite){
 45         try{
 46             //opponentID, isUserWhite, fen string (s),time game has taken
 47             //checks to see if file is empty
 48             FileReader read = new FileReader(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+session.getUsername()+".txt");
 49             BufferedReader buffRead = new BufferedReader(read);
 50             boolean isEmpty = buffRead.readLine()==null;buffRead.close();read.close();
 51             //writes new record to text file
 52             FileWriter write = new FileWriter(OfflinePlay.OFFLINE_GAMES_FOLDER+File.separator+session.getUsername()+".txt",true);
 53             BufferedWriter buffWrite = new BufferedWriter(write);
 54             if(isEmpty==false){
 55                 buffWrite.newLine();
 56             }
 57             buffWrite.write(opponentId+UserInfo.SPLIT_ITEM+String.valueOf(isUserWhite)+UserInfo.SPLIT_ITEM+OfflinePlay.NEW_BOARD+UserInfo.SPLIT_ITEM+"0");
 58             buffWrite.flush();write.flush();buffWrite.close();write.close();
 59         }catch(IOException e){
 60             JOptionPane.showMessageDialog(this,"An unexpected error has occured "+e,"Error",JOptionPane.OK_OPTION);
 61         }
 62     }

282     private void createGameBtnActionPerformed(java.awt.event.ActionEvent evt) {                                              
283         // creates a new game after checking that game settings have actually been selected;
284         //gets opponent's name
285         String opponentName=opponentNameTxt.getText();
286         //ensures that opponent name is not a computer id
287         if(opponentName.equals(OfflinePlay.LEVEL_ONE_COMP)||opponentName.equals(OfflinePlay.LEVEL_TWO_COMP)||opponentName.equals(OfflinePlay.LEVEL_THREE_COMP)||opponentName.equals(OfflinePlay.LEVEL_FOUR_COMP)||opponentName.equals(OfflinePlay.LEVEL_FIVE_COMP)||Validation.containsIllegalStrings(opponentName)){
288             JOptionPane.showMessageDialog(this,"Invalid opponent name. Please change the opponent name.","Error",JOptionPane.OK_OPTION);
289             return;
290         }
291         boolean isUserWhite=false;//stores which colour the user will play as
292         if(whiteBtn.isSelected()){
293             isUserWhite=true;
294         }else if(blackBtn.isSelected()){
295             isUserWhite=false;
296         }else if(randomBtn.isSelected()){
297             //randomly assings user a colour
298             if(Math.random()>=0.5){
299                 isUserWhite=true;
300             }else{
301                 isUserWhite=false;
302             }
303         }else{
304             //error message if no colour chosen
305             JOptionPane.showMessageDialog(this,"Please select a colour to play as.","Error",JOptionPane.OK_OPTION);
306             return;           
307         }
308         String gameTitle=gameTitleTxt.getText();
309         if(CreateOfflineGame.isGameTitleValid(session, gameTitle)==false||Validation.isPresent(gameTitle)==false){
310             JOptionPane.showMessageDialog(this,"Please select a valid and unique title.","Error",JOptionPane.OK_OPTION);
311             return;
312         }
313         int timeLeft=-100;
314         if(unlimitedBtn.isSelected()){
315             timeLeft=-100;
316         }else if(fiveMinutesBtn.isSelected()){
317             timeLeft=5*60;
318         }else if(tenMinutesBtn.isSelected()){
319             timeLeft =10*60;
320         }else{
321             JOptionPane.showMessageDialog(this,"Please select a time control setting.","Error",JOptionPane.OK_OPTION);
322             return;
323         }
324         CreateOfflineGame.addNewGameToFile(gameTitle, opponentName, isUserWhite, timeLeft, session);//writes record to file
325         //takes user to game
326         ChessBoardState[]state={new ChessBoardState(OfflinePlay.NEW_BOARD)};
327         ChessBoardOffline board = new ChessBoardOffline(session, opponentName, new ChessGame(state,false), isUserWhite, 0, timeLeft, timeLeft, gameTitle);
328         board.setVisible(true);
329         this.dispose();
330     }
