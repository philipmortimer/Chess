293     private void createNewLeaderBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                   
294         // appoints a new chess club leader
295         //asks for comfirmation before making the appointment
296         int confirm = JOptionPane.showConfirmDialog(this, "Are you sure you want to make somebody else the leader? This will mean that you will no longer be the leader.", "Proceed?", JOptionPane.YES_NO_CANCEL_OPTION);
297         if(confirm==JOptionPane.YES_OPTION){
298             //appoints somebody else as the chess club leader
299             String newLeader = JOptionPane.showInputDialog(this,"Please input the username of the new leader:");
300             if(newLeader==null){//if the user pressses cancel on the input dialog boc, nothing should occur
301                 return;
302             }
303             try{
304                 //performs validation
305                 if(ThreadClient.isUserNameAlreadyTaken(newLeader)==false){//checks to see that the username chosen is actually valid
306                     JOptionPane.showMessageDialog(this,"This user does not exist. Please input a real user.","Error",JOptionPane.OK_OPTION);
307                     return;
308                 }
309                 if(newLeader.equals(session.getUsername())){//if the new leader is just the same as the current leader, nothing needs to occur
310                     return;
311                 }
312                 ThreadClient.setUserAsClubLeader(newLeader);//sets the new leader
313                 session.setIsClubLeader(false);//updates the session data
314                 JOptionPane.showMessageDialog(this,"You are no longer the club leader.","Success",JOptionPane.OK_OPTION);
315                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();//takes the user back to the menu
316                 return;
317             }catch(Exception e){
318                 JOptionPane.showMessageDialog(this,"Error when communicating with the server. "+e,"Error",JOptionPane.OK_OPTION);
319                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();
320                 return;
321             }
322         }else{
323             return;
324         }   
325     }

327     private void viewLeagueTableBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                   
328         //takes the leader to a form that allows them to veiw the current club league table
329         try{
330             String[][]leagueTable=ThreadClient.getLeagueTable();//requests the league table from the server

358     private void changeContactDetailsBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                        
359         //attempts to change the contact details or save changes to contact details
360         if(areContactDetailsBeingEdited){
361             //attempts to save changes to server
362             //updates form to reflect the fact that the contact details are no longer activel being edited
363             areContactDetailsBeingEdited=false;
364             cancelChangesBtn.setVisible(false);cancelChangesBtn.setEnabled(false);
365             changeContactDetailsBtn.setText("Change Contact Details");
366             phoneNumberTxt.setEditable(false);
367             emailTxt.setEditable(false);
368             try{
369                 ThreadClient.setChessClubContactDetails(phoneNumberTxt.getText(), emailTxt.getText());//saves changes
370             }catch(Exception e){
371                 JOptionPane.showMessageDialog(this,"Error when communicating with the server: "+e,"Error",JOptionPane.OK_OPTION);
372                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();return;
373             }
374             phoneNumber=phoneNumberTxt.getText();email=emailTxt.getText();
375         }else{//sets it so that the leader can edit the contact details
376             //updates the form to allow the contact details to be changed
377             cancelChangesBtn.setVisible(true);cancelChangesBtn.setEnabled(true);
378             changeContactDetailsBtn.setText("Save Changes");
379             areContactDetailsBeingEdited=true;
380             phoneNumber=phoneNumberTxt.getText();email=emailTxt.getText();
381             phoneNumberTxt.setEditable(true);
382             emailTxt.setEditable(true);
383         }
384     }
