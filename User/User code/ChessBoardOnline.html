 348     public void updateBoard(){
 349         ChessBoardState state = game.getState(currentIndex);//gets state specified
 350         viewPreviousBoard.setEnabled(true);nextBoardBtn.setEnabled(true);viewPresentBoard.setEnabled(true);
 351         moveNumberLbl.setText("Move Number: "+String.valueOf(state.FULL_MOVE_NUMBER));
 352         if(currentIndex==0){//sets it so that no previous states can be accessed
 353             viewPreviousBoard.setEnabled(false);
 354         }
 355         if(currentIndex==game.getLengthOfStatesFoFar()-1){
 356             nextBoardBtn.setEnabled(false);//sets it so that view next board is not enabled
 357             viewPresentBoard.setEnabled(false);
 358         }
 359         //adds visual mark to display who's turn it is to play
 360         if(state.IS_BLACK_TURN_TO_PLAY_NEXT){
 361             isBlackTurnLbl.setText("*");isWhiteTurnLbl.setText("");
 362         }else{
 363             isBlackTurnLbl.setText("");isWhiteTurnLbl.setText("*");
 364         }
 365         if(game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT){
 366             blackTimeLbl.setText("*");whiteTiimeLbl.setText("");
 367         }else{
 368             blackTimeLbl.setText("");whiteTiimeLbl.setText("*");
 369         }
 370         int[][]board=state.getCopyOfBoard();//gets a copy of the game board
 371         if(isWhiteAtBottom==false){//checks to see whether board needs to be flipped
 372             //rotates board by 180 degrees
 373             rotateMatrix(board);
 374         }
 375         //disables resign button if game is played against computer and its the computers turn
 376         yCoordSquareClicked=-1;xCoordSquareClicked=-1;
 377         legalSquaresToMoveTo=new ArrayList<>();
 378         //loads the board and the corresponding images
 379         int yKing=-1;int xKing=-1;
 380         for(int y=0;y<8;y++){
 381             for(int x=0;x<8;x++){
 382                //sets square to the appropriate colour
 383                 if(y%2==x%2){
 384                     if(isDefaultLayout){
 385                         boardVisual[y][x].setBackground(Color.white);
 386                     }else{
 387                         boardVisual[y][x].setBackground(Color.decode("#F1D9C0"));  
 388                     }
 389                 }else{
 390                     if(isDefaultLayout){
 391                         boardVisual[y][x].setBackground(Color.black);
 392                     }else{
 393                         boardVisual[y][x].setBackground(Color.decode("#A97A65")); 
 394                     }  
 395                 }
 396                 //sets board labels to the appropriate piece image
 397                 switch (board[y][x]) {
 398                     case ChessBoardState.EMPTY_VALUE:
 399                         boardVisual[y][x].setIcon(null);
 400                         break;
 401                     case ChessBoardState.PAWN_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 402                         boardVisual[y][x].setIcon(imagePiece[11]);
 403                         break;
 404                     case ChessBoardState.PAWN_VALUE*ChessBoardState.BLACK_MULTIPLIER:
 405                         boardVisual[y][x].setIcon(imagePiece[5]);
 406                         break;
 407                     case ChessBoardState.ROOK_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 408                         boardVisual[y][x].setIcon(imagePiece[8]);
 409                         break;
 410                     case ChessBoardState.ROOK_VALUE*ChessBoardState.BLACK_MULTIPLIER:
 411                         boardVisual[y][x].setIcon(imagePiece[2]);
 412                         break;
 413                     case ChessBoardState.KNIGHT_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 414                         boardVisual[y][x].setIcon(imagePiece[10]);
 415                         break;
 416                     case ChessBoardState.KNIGHT_VALUE*ChessBoardState.BLACK_MULTIPLIER:
 417                         boardVisual[y][x].setIcon(imagePiece[4]);
 418                         break;
 419                     case ChessBoardState.BISHOP_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 420                         boardVisual[y][x].setIcon(imagePiece[9]);
 421                         break;
 422                     case ChessBoardState.BISHOP_VALUE*ChessBoardState.BLACK_MULTIPLIER:
 423                         boardVisual[y][x].setIcon(imagePiece[3]);
 424                         break;
 425                     case ChessBoardState.QUEEN_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 426                         boardVisual[y][x].setIcon(imagePiece[7]);
 427                         break;
 428                     case ChessBoardState.QUEEN_VALUE*ChessBoardState.BLACK_MULTIPLIER:
 429                         boardVisual[y][x].setIcon(imagePiece[1]);
 430                         break;
 431                     case ChessBoardState.KING_VALUE*ChessBoardState.WHITE_MULTIPLIER:
 432                         boardVisual[y][x].setIcon(imagePiece[6]);
 433                         if(!state.IS_BLACK_TURN_TO_PLAY_NEXT){
 434                             yKing=y;xKing=x;
 435                         }
 436                         break;
 437                     default:
 438                         //black king
 439                         boardVisual[y][x].setIcon(imagePiece[0]);
 440                         if(state.IS_BLACK_TURN_TO_PLAY_NEXT){
 441                             yKing=y;xKing=x;
 442                         }
 443                         break;
 444                 }
 445             }
 446         }
 447         //sets the king's square to be red if it is in check

 449             boardVisual[yKing][xKing].setBackground(Color.red);
 450         }

 548     private static void rotateMatrix(int[][] board) {
 549         int temp;//used as a buffer
 550         for (int i = 0; i <= 3; i++) {
 551             for (int j = 0; j < 8; j++) {
 552                 temp = board[i][j];
 553                 board[i][j] = board[7 - i ][7- j];
 554                 board[7-i][7- j] = temp;
 555             }
 556         }
