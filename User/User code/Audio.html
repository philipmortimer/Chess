15  * Plays desired sounds

18 public class Audio{
19     /**
20      * Plays the specified sound file. The sound method is threaded
21      * @param PATH The file name
22      */
23     public static synchronized void playSound(final String PATH) {
24        //creates new thread that plays the desired sound clip
25       new Thread(new Runnable() {
26           /**
27            * Plays the chosen audio clip
28            */
29         @Override
30         public void run() {
31           try {
32             Clip clip = AudioSystem.getClip();
33             AudioInputStream inputStream = AudioSystem.getAudioInputStream(
34               new File(PATH));
35             clip.open(inputStream);
36             clip.start(); 
37           } catch (Exception e) {
38             
39           }
40         }
41       }).start();
42     }
43 }
