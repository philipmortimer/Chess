 362     /**
 363      * This rotates a chess board array by 180 degrees.
 364      * @param board The board (must be an 8*8 array for this rotation method to work)
 365      */
 366     private static void rotateMatrix(int[][] board) {
 367         int temp;//used as a buffer
 368         //rotates 8*8 board
 369         for (int i = 0; i <= 3; i++) {
 370             for (int j = 0; j < 8; j++) {
 371                 temp = board[i][j];
 372                 board[i][j] = board[7 - i ][7- j];
 373                 board[7-i][7- j] = temp;
 374             }
 375         }
 376     }

 849     /**
 850      * Checks to see if current board is in a terminal state. If it is, the game result is announced and then the system terminates
 851      */
 852     public void checkForTerminalState(){
 853         if(game.getAllLegalBoardsFromState(game.getCurrentState()).size()==0){//checks to see if there are any legal moves from the current board state - if there are not then the game must be over
 854             timer.stop();//stops the timers
 855             if(game.getCurrentState().getIsDraw()){
 856                 //displays that the game was a draw
 857                 JOptionPane.showMessageDialog(this, "The game was a draw!","Draw!",JOptionPane.OK_OPTION);
 858                 deleteGameAndReturnToMenu();
 859                 return;
 860             }
 861             if(ChessGame.isInCheck(!game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT, new ChessBoardState(game.getCurrentState().BOARD, game.getCurrentState().CASTLING_RIGHTS, !game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT, game.getCurrentState().FULL_MOVE_NUMBER, game.getCurrentState().HALF_MOVE_CLOCK, game.getCurrentState().SQUARE_THAT_CAN_BE_CAPTURED_EN_PASSANT))){
 862                 //if the user is in check, then the outcome of the game must be checkmate
 863                 if(game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT){
 864                     JOptionPane.showMessageDialog(this, "White wins!","Checkmate!",JOptionPane.OK_OPTION);
 865                     deleteGameAndReturnToMenu();
 866                     return;
 867                 }else{
 868                     JOptionPane.showMessageDialog(this, "Black wins.","Checkmate!",JOptionPane.OK_OPTION);
 869                     deleteGameAndReturnToMenu();
 870                     return;
 871                 }
 872             }
 873             JOptionPane.showMessageDialog(this, "The game was a draw!","Draw!",JOptionPane.OK_OPTION);
 874             deleteGameAndReturnToMenu();
 875             return;
 876         }
 877     }

 919     /**
 920      * Handles when a square is clicked. This includes highlighting the square and all available moves if appropriate.
 921      * @param y The y index of the square
 922      * @param x The x index of the square
 923      */
 924     private void squareClicked(int y,int x){
 925         //if opponent is a computer and its the computer's turn to play, no squares should be highlighted
 926         if((computerLevel!=-1)&&((isUserWhite&&game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT)||(!isUserWhite&&!game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT))){
 927             return;
 928         }
 929         if(currentIndex!=game.getLengthOfStatesFoFar()-1){//if the board is not the current board, no action is needed
 930             return;
 931         }
 932         int yBoard=y;int xBoard=x;//stores the y and x coords regarding to the actual board array stored
 933         if(isWhiteAtBottom==false){//accounts for a flipped board
 934             yBoard=7-y;xBoard=7-x;
 935         }
 936         //updateBoard();
 937         //checks to see if square clicked is a legal square to move to. If no square has been clicked, the array is of length 0
 938         for(int move=0;move<legalSquaresToMoveTo.size();move=move+2){
 939             if(legalSquaresToMoveTo.get(move)==yBoard&&legalSquaresToMoveTo.get(move+1)==xBoard){//if square choosen is a move, it updates game object and board
 940                 ArrayList<ChessBoardState>states=ChessGame.getAllLegalBoardsFromMove(yCoordSquareClicked, xCoordSquareClicked, yBoard, xBoard, game.getCurrentState());
 941                 if(states.size()==1){
 942                     game.addState(states.get(0));
 943                 }else{//used to allow user to make choice about pawn promotion
 944                     Object[] options = {"Queen","Rook","Bishop","Knight"};//stores the possible pieces to promote
 945                     int choice=JOptionPane.showOptionDialog(this,"Please select piece to promote pawn to.","Piece",JOptionPane.YES_NO_CANCEL_OPTION,JOptionPane.QUESTION_MESSAGE,null,options,options[0]);
 946                     if(choice==-1){//if no option is seleted, the queen is chosen
 947                         game.addState(states.get(0));
 948                     }else{
 949                         game.addState(states.get(choice));
 950                     }
 951                 }
 952                 //updates board following the move having been made
 953                 currentIndex++;
 954                 updateBoard();
 955                 playMoveMadeSound();
 956                 checkForTerminalState();
 957                 getComputerToMoveIfAppropriate();
 958                 return;
 959             }
 960         }
 961         //if invalid square is selected, nothing occurrs
 962         if((game.getCurrentState().BOARD[yBoard][xBoard]>=0&&game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT)||(game.getCurrentState().BOARD[yBoard][xBoard]<=0&&!game.getCurrentState().IS_BLACK_TURN_TO_PLAY_NEXT)){
 963             return;
 964         }
