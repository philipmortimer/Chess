 54             public Component getComponentBefore(Container aContainer, Component aComponent) {
 55                 currentCompIndex--;
 56                 if(currentCompIndex<0){//wraps the array so that it goes to the end
 57                     currentCompIndex=COMPS.length-1;

548      * Attempts to create an account after performing validation on the appropriate fields
549      * @param evt The button being pressed
550      */
551     private void createAccountBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                 
552         //validates the account details to ensure they are valid. Once they have been validated, the account is created
553         //presence check
554         if(!(Validation.isPresent(usernameTxt.getText())&& Validation.isPresent(emailAddressTxt.getText())&& Validation.isPresent(new String(passwordField.getPassword()))&&Validation.isPresent(new String(confirmPasswordField.getPassword())))){
555             JOptionPane.showMessageDialog(this, "Username, email address, password and confirm password must not be left blank.","Invalid input",JOptionPane.OK_OPTION);
556             return;
557         }
558         if(Validation.containsIllegalStrings(usernameTxt.getText()+emailAddressTxt.getText()+new String(passwordField.getPassword())+new String(confirmPasswordField.getPassword())+firstNameTxt.getText()+surnameTxt.getText()+postcodeTxt.getText()+addressLineOneTxt.getText()
559                 +addressLineTwoTxt.getText()+addressLineThreeTxt.getText()+addressLineFourTxt.getText()+discordNumberTxt.getText()+discordUsernameTxt.getText()+dateOfBirthTxt.getText()+phoneNumberTxt.getText())){//performs a look up check to see whether any field contains an illegal character
560             JOptionPane.showMessageDialog(this, "One or more fields contain an illegal character sequence. Here is a list of illegal caharacter sequences: "+Validation.getListOfIllegalStrings(),"Invalid Input",JOptionPane.OK_OPTION);
561             return;
562         }
563         //check username inputs
564         if(!Validation.isAppropriateLength(15, usernameTxt.getText())){//checks to see if username is 15 charcters or less - length check
565             JOptionPane.showMessageDialog(this, "The username can not exceed 15 characters in length","Invalid Input",JOptionPane.OK_OPTION);
566             return;
567         }
568         if(usernameTxt.getText().contains("@") || usernameTxt.getText().contains(" ")){//format check
569             JOptionPane.showMessageDialog(this, "The username can not contain the @ symbol OR a space.","Invalid Input",JOptionPane.OK_OPTION);
570             return;
571         }
572         //checks email address
573         if(!emailAddressTxt.getText().contains("@") || emailAddressTxt.getText().contains(" ")){
574             JOptionPane.showMessageDialog(this, "The email address must contain the @ symbol and no spaces.","Invalid Input",JOptionPane.OK_OPTION);
575             return;            
576         }
577         //length check
578         if(!Validation.isAppropriateLength(39, emailAddressTxt.getText())){//email address must not be longer than 39 characters
579             JOptionPane.showMessageDialog(this, "The email address must not be longer than 39 characters.","Invalid Input",JOptionPane.OK_OPTION);
580             return;               
581         }
582         if(!new String(passwordField.getPassword()).equals(new String (confirmPasswordField.getPassword()))){//checks that both passwords match - double entry verification validation
583             JOptionPane.showMessageDialog(this, "The two passwords do not match.","Invalid Input",JOptionPane.OK_OPTION);
584             return;                   
585         }
586         if(new String(passwordField.getPassword()).length()<5 || new String(passwordField.getPassword()).length()>20){//length check
587             JOptionPane.showMessageDialog(this, "The password must be between 5-20 characters in length (inclusive).","Invalid Input",JOptionPane.OK_OPTION);
588             return;                     
589         }
590         if(Validation.isPresent(firstNameTxt.getText())){//only performs validation if first name is not empty
591             if(!Validation.isAppropriateLength(35, firstNameTxt.getText())){//length check - first name can't be more than 35 characters
592                 JOptionPane.showMessageDialog(this, "The first name can not be longer than 35 characters","Invalid Input",JOptionPane.OK_OPTION);
593                 return;                     
594             }
595             if(!Validation.consistsOnlyOfLetters(firstNameTxt.getText())){
596                 JOptionPane.showMessageDialog(this, "The first name can only contain characters that are in the english alphabet","Invalid Input",JOptionPane.OK_OPTION);
597                 return;                  
598             }
599         }
600         if(Validation.isPresent(surnameTxt.getText())){//only performs validation if  surname is not empty
601             if(!Validation.isAppropriateLength(40, surnameTxt.getText())){//length check - surname can't be more than 40 characters
602                 JOptionPane.showMessageDialog(this, "The surname can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
603                 return;                     
604             }
605             if(!Validation.consistsOnlyOfLetters(surnameTxt.getText())){
606                 JOptionPane.showMessageDialog(this, "The surname can only contain characters that are in the english alphabet","Invalid Input",JOptionPane.OK_OPTION);
607                 return;                  
608             }
609         }
610         if(Validation.isPresent(postcodeTxt.getText())){//only performs validation is postcode is not left empty
611             if(!Validation.isValidPostcode(postcodeTxt.getText())){
612                 JOptionPane.showMessageDialog(this, "The postcode must be between 6-8 characters long, contain exactly one space and have all non-space characters capitalised","Invalid Input",JOptionPane.OK_OPTION);
613                 return;      
614             }
615         }
616         if(!Validation.isAppropriateLength(40, addressLineOneTxt.getText())){//length check
617             JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
618             return;              
619         }
620         if(!Validation.isAppropriateLength(40, addressLineTwoTxt.getText())){//length check
621             JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
622             return;              
623         }
624         if(!Validation.isAppropriateLength(40, addressLineThreeTxt.getText())){//length check
625             JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
626             return;              
627         }
628         if(!Validation.isAppropriateLength(40, addressLineFourTxt.getText())){//length check
629             JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
630             return;              
631         }
632         if(!Validation.isAppropriateLength(32, discordUsernameTxt.getText())){
633             JOptionPane.showMessageDialog(this, "The Discord username can not be longer than 32 characters","Invalid Input",JOptionPane.OK_OPTION);
634             return;              
635         }
636         if(Validation.isPresent(discordNumberTxt.getText())){
637             if(!Validation.isInteger(discordNumberTxt.getText())){
638                 JOptionPane.showMessageDialog(this, "The Discord number must be an integer","Invalid Input",JOptionPane.OK_OPTION);
639                 return;            
640             }
641             if(!Validation.isAppropriateLength(4, discordNumberTxt.getText())){
642                 JOptionPane.showMessageDialog(this, "The Discord number must be no more than 4 characters","Invalid Input",JOptionPane.OK_OPTION);
643                 return;                       
644             }
645             if(!Validation.rangeCheck(0, 9999, Integer.parseInt(discordNumberTxt.getText()))){//range check
646                 JOptionPane.showMessageDialog(this, "The Discord number must be between 0000-9999","Invalid Input",JOptionPane.OK_OPTION);
647                 return;                   
648             }
649         }
650         if(Validation.isPresent(dateOfBirthTxt.getText())){
651             if(!Validation.formatCheckDateOfBirth(dateOfBirthTxt.getText())){
652                 JOptionPane.showMessageDialog(this, "The date of birth must be a valid date of birth in the form dd/mm/yyyy","Invalid Input",JOptionPane.OK_OPTION);
653                 return;   
654             }
655         }
656         if(Validation.isPresent(phoneNumberTxt.getText())){
657             if(!Validation.isAppropriateLength(20, phoneNumberTxt.getText())){
658                 JOptionPane.showMessageDialog(this, "Phone number can not be longer than 20 characters.","Invalid Input",JOptionPane.OK_OPTION);
659                 return;                   
660             }
661             if(!Validation.onlyContainsPhoneNumberCharacters(phoneNumberTxt.getText())){
662                 JOptionPane.showMessageDialog(this, "Phone number can only contain plusses, spaces and integers.","Invalid Input",JOptionPane.OK_OPTION);
663                 return;                     
664             }
665         }
666         //lookup check to ensure that both email address and username are not already taken
667         try{
668             if(ThreadClient.isUserNameAlreadyTaken(usernameTxt.getText())){
669                 JOptionPane.showMessageDialog(this, "The username is already taken. Plesae try a different username.","Invalid Input",JOptionPane.OK_OPTION);
670                 return;
671             }
672             if(ThreadClient.isEmailAddressAlreadyTaken(emailAddressTxt.getText())){
673                 JOptionPane.showMessageDialog(this, "The email is already taken. Plesae try a different email address.","Invalid Input",JOptionPane.OK_OPTION);                
674                 return;
675             }
676         }catch(Exception e){
677             JOptionPane.showMessageDialog(this, "An error occured: "+e.getMessage()+". This is almost certainly becuase something is wrong with your internet connection or the server.","Network Error",JOptionPane.OK_OPTION);
678             return;
679         }
680         try{
681             //creates the account once the validation has been performed
682             ThreadClient.createAccount(new String[]{usernameTxt.getText(),new String(passwordField.getPassword()),firstNameTxt.getText(),surnameTxt.getText(),postcodeTxt.getText(),addressLineOneTxt.getText(),addressLineTwoTxt.getText(),addressLineThreeTxt.getText(),addressLineFourTxt.getText(),discordUsernameTxt.getText(),discordNumberTxt.getText(),dateOfBirthTxt.getText(),phoneNumberTxt.getText(),"false","false","1000","false"},emailAddressTxt.getText());
683             JOptionPane.showMessageDialog(this,"Account succesfully created.","Success", JOptionPane.OK_OPTION);
684             LogIn log = new LogIn();//takes the user to the log in screen once the account has been made
685             log.setVisible(true);this.dispose();
686             return;
687         }catch(Exception e){
688             JOptionPane.showMessageDialog(this, "An error occured: "+e.getMessage()+". This is almost certainly becuase something is wrong with your internet connection or the server.","Network Error",JOptionPane.OK_OPTION);
689             return;
690         }
691     }
