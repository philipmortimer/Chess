232     private void leaveClubBtnActionPerformed(java.awt.event.ActionEvent evt) {                                             
233         // lets the user leave the chess club after confirming they wish to do so
234         int confirm = JOptionPane.showConfirmDialog(this, "Are you sure you would like to leave the chess club?", "Quit?", JOptionPane.YES_NO_CANCEL_OPTION);
235         if(confirm==JOptionPane.YES_OPTION){
236             //removes user from the chess club
237             try{
238                 ThreadClient.removeUserFromChessClub(session.getUsername());
239                 session.setIsClubMember(false);
240                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();
241                 return;
242             }catch(Exception e){
243                 JOptionPane.showMessageDialog(this,"Error when communicating with the server. "+e,"Error",JOptionPane.OK_OPTION);
244                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();
245                 return;
246             }
247         }else{
248             return;
249         }   
250     }

252     private void viewLeagueTableBtnActionPerformed(java.awt.event.ActionEvent evt) {                                                   
253         //takes the user to a form to view the current league table
254         try{
255             String[][]leagueTable=ThreadClient.getLeagueTable();//requests the league table from the server
256             if(leagueTable.length==0){
257                 //if the no fixtures have been generated, then the league table requires all the chess club members as an input instead
258                 String[]allPlayers=ThreadClient.getMembersOfChessClub();
259                 LeagueTable tab = new LeagueTable(session, allPlayers);
260                 tab.setVisible(true);this.dispose();return;
261             }
262             LeagueTable tab = new LeagueTable(session, leagueTable);
263             tab.setVisible(true);this.dispose();return;
264         }catch(Exception e){
265             JOptionPane.showMessageDialog(this,"Error when communicating with the server: "+e,"Error",JOptionPane.OK_OPTION);
266             return;
267         }
268     }
