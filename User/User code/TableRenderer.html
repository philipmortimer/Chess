 8 import java.awt.Color;
 9 import java.awt.Component;
10 import javax.swing.JLabel;
11 import javax.swing.JTable;
12 import javax.swing.table.DefaultTableCellRenderer;
13 
14 /**
15  * This creates a custom table renderer object to render the online games table.
16  * @author mortimer
17  */
18 public class TableRenderer extends DefaultTableCellRenderer{
19     private boolean rowOneHighlighted=false;//determines the colour of row one
20     private boolean rowTwoHighlighted=false;//determines the colour of row two
21     private boolean rowThreeHighlighted=false;//determines the colour of row three
22     private static final Color HIGHLIGHT_COLOR=Color.WHITE;//stores the color used for highlighted rows
23     private static final Color JLABEL_COLOUR=Color.LIGHT_GRAY;//stores the default JLabel rows.
24     private final OnlineMenu ONLINE;//stores the online menu in which the table is locared
25     /**
26      * Creates a custom table renderer to render the online game table.
27      * @param online The online menu form in which the online table is in.
28      */
29     public TableRenderer(OnlineMenu online){
30         ONLINE=online;//stores the online menu form
31     }
32     /**
33      * Updates the variables that change the table row colour
34      * @param rowOneIsHighlighted Whether row one should be highlighted
35      * @param rowTwoIsHighlighted Whether row two should be highlighted
36      * @param rowThreeIsHighlighted Whether row three should be highlighted
37      */
38     public void updateRowColour(boolean rowOneIsHighlighted,boolean rowTwoIsHighlighted,boolean rowThreeIsHighlighted){
39         rowOneHighlighted=rowOneIsHighlighted;rowTwoHighlighted=rowTwoIsHighlighted;rowThreeHighlighted=rowThreeIsHighlighted;//sets object variables equal to the passed variables
40     }
41     /**
42      * Returns the customised component to be rendered
43      * @param table The table
44      * @param value The value of the component
45      * @param isSelected Whether the component is selected
46      * @param hasFocus Whether the component is focused
47      * @param row The row of the component
48      * @param col The column of the table
49      * @return The component
50      */
51     @Override
52     public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int col) {
53         JLabel label = (JLabel) super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, col);//gets JLabel for the row + col chosen
54         //Get the status for the current row and sets the colour of the JLabel 
55         Color cellColour = JLABEL_COLOUR;
56         switch (row) {
57             case 0:
58                 if (rowOneHighlighted) {
59                     cellColour = HIGHLIGHT_COLOR;
60                 }
61                 break;
62             case 1:
63                 if (rowTwoHighlighted) {
64                     cellColour = HIGHLIGHT_COLOR;
65                 }
66                 break;
67             default:
68                 if (rowThreeHighlighted) {
69                     cellColour = HIGHLIGHT_COLOR;
70                 }
71                 break;
72         }
73         if (isSelected) {
74             if (cellColour == JLABEL_COLOUR) {
75                 ONLINE.setPlayButtonEnabled(false);
76                 ONLINE.setDeleteButtonEnabled(!table.getValueAt(row, 0).equals(""));
77             } else {
78                 ONLINE.setPlayButtonEnabled(true);
79                 ONLINE.setDeleteButtonEnabled(false);
80                 cellColour = table.getSelectionBackground();
81             }
82         }
83         label.setBackground(cellColour);//sets colour to chosen colour
84         return label;
85     }
86 }
