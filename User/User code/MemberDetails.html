 51     private void loadFormBasedOnIndex(){
 52         clearForm();//clears the form
 53         //determines whether the buttons to naviagte between users should be enabled or not.
 54         prevBtn.setEnabled(true);nextBtn.setEnabled(true);
 55         if(clubMembersIndex==0){
 56             prevBtn.setEnabled(false);
 57         }
 58         if(clubMembersIndex==clubMembersInfo.length-1){
 59             nextBtn.setEnabled(false);
 60         }
 61         //ensuers that the leader can't remove themselves from the club
 62         removeBtn.setEnabled(true);
 63         if(clubMembersInfo[clubMembersIndex][0].equals(session.getUsername())){
 64             removeBtn.setEnabled(false);
 65         }
 66         //Username, password, first name, surname, postcode, addressline one, address line two, address line three, address line four, discord username, discord number, date of birth, phone number, isClubLeader, isClubMember, ELO, hasUserRequestedToJoinClub
 67         //sets the value of the various fields based on the user informtion provided
 68         usernameTxt.setText(clubMembersInfo[clubMembersIndex][0]);
 69         emailAddressTxt.setText(emails[clubMembersIndex]);
 70         firstNameTxt.setText(clubMembersInfo[clubMembersIndex][2]);surnameTxt.setText(clubMembersInfo[clubMembersIndex][3]);
 71         postcodeTxt.setText(clubMembersInfo[clubMembersIndex][4]);addressLineOneTxt.setText(clubMembersInfo[clubMembersIndex][5]);
 72         addressLineTwoTxt.setText(clubMembersInfo[clubMembersIndex][6]);addressLineThreeTxt.setText(clubMembersInfo[clubMembersIndex][7]);
 73         addressLineFourTxt.setText(clubMembersInfo[clubMembersIndex][8]);discordUsernameTxt.setText(clubMembersInfo[clubMembersIndex][9]);
 74         discordNumberTxt.setText(clubMembersInfo[clubMembersIndex][10]);dateOfBirthTxt.setText(clubMembersInfo[clubMembersIndex][11]);
 75         phoneNumberTxt.setText(clubMembersInfo[clubMembersIndex][12]);eloTxt.setText(clubMembersInfo[clubMembersIndex][15]);
 76     }

539     private void editBtnActionPerformed(java.awt.event.ActionEvent evt) {                                        
540         //either saves changes made to a user's profile or enters editing mode for a given user
541         if(isUserInfoBeingEdited){
542             //saves changes to user to server
543             //checks to see that changes are valid
544             //presence check
545             if(Validation.containsIllegalStrings(usernameTxt.getText()+emailAddressTxt.getText()+firstNameTxt.getText()+surnameTxt.getText()+postcodeTxt.getText()+addressLineOneTxt.getText()
546                     +addressLineTwoTxt.getText()+addressLineThreeTxt.getText()+addressLineFourTxt.getText()+discordNumberTxt.getText()+discordUsernameTxt.getText()+dateOfBirthTxt.getText()+phoneNumberTxt.getText())){//performs a look up check to see whether any field contains an illegal character
547                 JOptionPane.showMessageDialog(this, "One or more fields contain an illegal character sequence. Here is a list of illegal caharacter sequences: "+Validation.getListOfIllegalStrings(),"Invalid Input",JOptionPane.OK_OPTION);
548                 return;
549             }
550             if(Validation.isPresent(firstNameTxt.getText())){//only performs validation if first name is not empty
551                 if(!Validation.isAppropriateLength(35, firstNameTxt.getText())){//length check - first name can't be more than 35 characters
552                     JOptionPane.showMessageDialog(this, "The first name can not be longer than 35 characters","Invalid Input",JOptionPane.OK_OPTION);
553                     return;                     
554                 }
555                 if(!Validation.consistsOnlyOfLetters(firstNameTxt.getText())){
556                     JOptionPane.showMessageDialog(this, "The first name can only contain characters that are in the english alphabet","Invalid Input",JOptionPane.OK_OPTION);
557                     return;                  
558                 }
559             }
560             if(Validation.isPresent(surnameTxt.getText())){//only performs validation if  surname is not empty
561                 if(!Validation.isAppropriateLength(40, surnameTxt.getText())){//length check - surname can't be more than 40 characters
562                     JOptionPane.showMessageDialog(this, "The surname can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
563                     return;                     
564                 }
565                 if(!Validation.consistsOnlyOfLetters(surnameTxt.getText())){
566                     JOptionPane.showMessageDialog(this, "The surname can only contain characters that are in the english alphabet","Invalid Input",JOptionPane.OK_OPTION);
567                     return;                  
568                 }
569             }
570             if(Validation.isPresent(postcodeTxt.getText())){//only performs validation is postcode is not left empty
571                 if(!Validation.isValidPostcode(postcodeTxt.getText())){
572                     JOptionPane.showMessageDialog(this, "The postcode must be between 6-8 characters long, contain exactly one space and have all non-space characters capitalised","Invalid Input",JOptionPane.OK_OPTION);
573                     return;      
574                 }
575             }
576             if(!Validation.isAppropriateLength(40, addressLineOneTxt.getText())){//length check
577                 JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
578                 return;              
579             }
580             if(!Validation.isAppropriateLength(40, addressLineTwoTxt.getText())){//length check
581                 JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
582                 return;              
583             }
584             if(!Validation.isAppropriateLength(40, addressLineThreeTxt.getText())){//length check
585                 JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
586                 return;              
587             }
588             if(!Validation.isAppropriateLength(40, addressLineFourTxt.getText())){//length check
589                 JOptionPane.showMessageDialog(this, "Each address line can not be longer than 40 characters","Invalid Input",JOptionPane.OK_OPTION);
590                 return;              
591             }
592             if(!Validation.isAppropriateLength(32, discordUsernameTxt.getText())){
593                 JOptionPane.showMessageDialog(this, "The Discord username can not be longer than 32 characters","Invalid Input",JOptionPane.OK_OPTION);
594                 return;              
595             }
596             if(Validation.isPresent(discordNumberTxt.getText())){
597                 if(!Validation.isInteger(discordNumberTxt.getText())){
598                     JOptionPane.showMessageDialog(this, "The Discord number must be an integer","Invalid Input",JOptionPane.OK_OPTION);
599                     return;            
600                 }
601                 if(!Validation.isAppropriateLength(4, discordNumberTxt.getText())){
602                     JOptionPane.showMessageDialog(this, "The Discord number must be no more than 4 characters","Invalid Input",JOptionPane.OK_OPTION);
603                     return;                       
604                 }
605                 if(!Validation.rangeCheck(0, 9999, Integer.parseInt(discordNumberTxt.getText()))){//range check
606                     JOptionPane.showMessageDialog(this, "The Discord number must be between 0000-9999","Invalid Input",JOptionPane.OK_OPTION);
607                     return;                   
608                 }
609             }
610             if(Validation.isPresent(dateOfBirthTxt.getText())){
611                 if(!Validation.formatCheckDateOfBirth(dateOfBirthTxt.getText())){
612                     JOptionPane.showMessageDialog(this, "The date of birth must be a valid date of birth in the form dd/mm/yyyy","Invalid Input",JOptionPane.OK_OPTION);
613                     return;   
614                 }
615             }
616             if(Validation.isPresent(phoneNumberTxt.getText())){
617                 if(!Validation.isAppropriateLength(20, phoneNumberTxt.getText())){
618                     JOptionPane.showMessageDialog(this, "Phone number can not be longer than 20 characters.","Invalid Input",JOptionPane.OK_OPTION);
619                     return;                   
620                 }
621                 if(!Validation.onlyContainsPhoneNumberCharacters(phoneNumberTxt.getText())){
622                     JOptionPane.showMessageDialog(this, "Phone number can only contain plusses, spaces and integers.","Invalid Input",JOptionPane.OK_OPTION);
623                     return;                     
624                 }
625             }
626             //saves changes having validated them
627 //Username 0, password 1, first name 2, surname 3, postcode 4, addressline one 5, address line two 6, address line three 7, address line four 8, discord username 9, discord number 10, date of birth 11, phone number 12, isClubLeader 13, isClubMember 14, ELO 15, hasUserRequestedToJoinClub 16
628             String[]newRecord=new String[]{usernameTxt.getText(),clubMembersInfo[clubMembersIndex][1],firstNameTxt.getText(),surnameTxt.getText(),postcodeTxt.getText(),addressLineOneTxt.getText(),addressLineTwoTxt.getText(),addressLineThreeTxt.getText(),addressLineFourTxt.getText(),discordUsernameTxt.getText(),discordNumberTxt.getText(),dateOfBirthTxt.getText(),phoneNumberTxt.getText(),clubMembersInfo[clubMembersIndex][13],clubMembersInfo[clubMembersIndex][14],clubMembersInfo[clubMembersIndex][15],clubMembersInfo[clubMembersIndex][16]};
629             for(int x=0;x<UserInfo.WIDTH_OF_USER_DETAILS_FILE;x++){
630                 clubMembersInfo[clubMembersIndex][x]=newRecord[x];
631             }
632             try{
633                 ThreadClient.updateUserInfo(newRecord);//changes are saved
634             }catch(Exception e){
635                 JOptionPane.showMessageDialog(this,"Error when communicating with the server: "+e,"Error",JOptionPane.OK_OPTION);
636                 Menu menu = new Menu(session);menu.setVisible(true);this.dispose();return;
637             }
638             isUserInfoBeingEdited=false;
639             //updates components and exits editing mode
640             prevBtn.setEnabled(true);nextBtn.setEnabled(true);
641             removeBtn.setEnabled(true);
642             firstNameTxt.setEditable(false);surnameTxt.setEditable(false);
643             postcodeTxt.setEditable(false);
644             addressLineOneTxt.setEditable(false);addressLineTwoTxt.setEditable(false);addressLineThreeTxt.setEditable(false);
645             addressLineFourTxt.setEditable(false);discordUsernameTxt.setEditable(false);discordNumberTxt.setEditable(false);
646             dateOfBirthTxt.setEditable(false);phoneNumberTxt.setEditable(false);
647             loadFormBasedOnIndex();//updates form values
648             editBtn.setText("Edit Details");
649             removeBtn.setText("Remove Member");
650             removeBtn.setBackground(Color.red);
651         }else{
652             //puts form in editing mode - chaning looks and feel of components
653             editBtn.setText("Save Changes");
654             removeBtn.setText("Cancel Changes");
655             isUserInfoBeingEdited=true;
656             prevBtn.setEnabled(false);nextBtn.setEnabled(false);
657             removeBtn.setEnabled(true);removeBtn.setBackground(nextBtn.getBackground());

663             dateOfBirthTxt.setEditable(true);phoneNumberTxt.setEditable(true);
664         }

679     private void removeBtnActionPerformed(java.awt.event.ActionEvent evt) {                                          
680         //removes members or cancels changes made when editing
681         if(isUserInfoBeingEdited){
682             //cancles changes made whilst editing and reverst form to the state it was in before editing mode was entered
683             isUserInfoBeingEdited=false;
684             prevBtn.setEnabled(true);nextBtn.setEnabled(true);
685             removeBtn.setEnabled(true);
686             firstNameTxt.setEditable(false);surnameTxt.setEditable(false);
687             postcodeTxt.setEditable(false);
688             addressLineOneTxt.setEditable(false);addressLineTwoTxt.setEditable(false);addressLineThreeTxt.setEditable(false);
689             addressLineFourTxt.setEditable(false);discordUsernameTxt.setEditable(false);discordNumberTxt.setEditable(false);
690             dateOfBirthTxt.setEditable(false);phoneNumberTxt.setEditable(false);
691             loadFormBasedOnIndex();
692             editBtn.setText("Edit Details");
693             removeBtn.setText("Remove Member"); 
694             removeBtn.setBackground(Color.red);
695             return;
696         }else{
697             //removes a member from the chess club after asking for confirmation
698             int confirm = JOptionPane.showConfirmDialog(this, "Are you sure you would like to remove the member from the Chess Club?", "Remove?", JOptionPane.YES_NO_CANCEL_OPTION);
699             if(confirm==JOptionPane.YES_OPTION){
700                 try{
701                     ThreadClient.removeUserFromChessClub(clubMembersInfo[clubMembersIndex][0]);//removes user
702                 }catch(Exception e){
703                     JOptionPane.showMessageDialog(this,"Error when communicating with the server: "+e,"Error",JOptionPane.OK_OPTION);
704                     Menu menu = new Menu(session);menu.setVisible(true);this.dispose();return;
705                 }
706                 //removes user info from array
707                 String[][]newClubMemberInfo=new String[clubMembersInfo.length-1][UserInfo.WIDTH_OF_USER_DETAILS_FILE];
708                 for(int i=0;i<clubMembersIndex;i++){
709                     for(int x=0;x<UserInfo.WIDTH_OF_USER_DETAILS_FILE;x++){
710                         newClubMemberInfo[i][x]=clubMembersInfo[i][x];
711                     }
712                 }
713                 for(int i=clubMembersIndex+1;i<clubMembersInfo.length;i++){
714                     for(int x=0;x<UserInfo.WIDTH_OF_USER_DETAILS_FILE;x++){
715                         newClubMemberInfo[i-1][x]=clubMembersInfo[i][x];
716                     }                    
717                 }
718                 //removes email from array
719                 String[]newEmail=new String[emails.length-1];
720                 for(int i=0;i<clubMembersIndex;i++){
721                     newEmail[i]=emails[i];
722                 }
723                 for(int i=clubMembersIndex+1;i<emails.length;i++){
724                     newEmail[i-1]=emails[i];
725                 }
726                 emails=newEmail;
727                 clubMembersInfo=newClubMemberInfo;
728                 clubMembersIndex=0;
729                 loadFormBasedOnIndex();//loads updated form without removed member
730             }
731         }
732     }
