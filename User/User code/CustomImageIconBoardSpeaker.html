 8 import java.awt.Component;
 9 import java.awt.Graphics;
10 import java.awt.Graphics2D;
11 import java.awt.Image;
12 import java.awt.RenderingHints;
13 import java.awt.image.BufferedImage;
14 import javax.swing.ImageIcon;
15 
16 /**
17  * A custom image icon class that ensures the image icon of the speaker is of high quality
18  * @author mortimer
19  */
20 public class CustomImageIconBoardSpeaker extends ImageIcon {
21     /**
22      * Creates an image icon for the speaker which is of high visual quality.
23      * @param f The image of the speaker (which must be 15 by 15 pixels).
24      */
25     public CustomImageIconBoardSpeaker(Image f) {
26       super(f);
27       //sets a high render quality of the ImageIcon
28       BufferedImage i= new BufferedImage(15, 15, BufferedImage.TYPE_INT_ARGB);
29       //sets the image to a high rendering quality
30       Graphics2D g2d = (Graphics2D)i.getGraphics();
31                 g2d.setRenderingHint(RenderingHints.KEY_ALPHA_INTERPOLATION, RenderingHints.VALUE_ALPHA_INTERPOLATION_QUALITY);
32                 g2d.setRenderingHint(RenderingHints.KEY_COLOR_RENDERING, RenderingHints.VALUE_COLOR_RENDER_QUALITY);
33                 g2d.setRenderingHint(RenderingHints.KEY_DITHERING, RenderingHints.VALUE_DITHER_ENABLE);
34                 g2d.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS, RenderingHints.VALUE_FRACTIONALMETRICS_ON);
35                 g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
36                 g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
37                 g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
38                 
39       g2d.drawImage(getImage(), 0, 0, 15, 15, null);
40       setImage(i);
41     }
42     /**
43      * Gets the height of the icon.
44      * @return The height of the icon  (which is always 15)
45      */
46     @Override
47     public int getIconHeight() {
48       return 15;
49     }
50     /**
51      * Gets the width of the icon.
52      * @return The width of the icon  (which is always 15)
53      */
54     @Override
55     public int getIconWidth() {
56       return 15;
57     }
58     /**
59      * Paints the icon
60      * @param c The component
61      * @param g The graphics
62      * @param x The x coordinate
63      * @param y The y coordinate
64      */
65     @Override
66     public void paintIcon(Component c, Graphics g, int x, int y) {
67       g.drawImage(getImage(), x, y, c);
68     }   
69 }
