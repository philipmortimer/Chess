191     private void chessClubBtnActionPerformed(java.awt.event.ActionEvent evt) {                                             
192         // takes the user to the west cross chess club menu. This form differs depending on what type of user is attempting to access the form
193         if(session.getIsGuest()){//takes the user to the chess club form for guests
194             WestCrossChessClubGuest chessClubGuest = new WestCrossChessClubGuest(session);
195             chessClubGuest.setVisible(true);
196             this.dispose();
197         }else if(session.getIsClubMember()==false){//takes the user to the chess club form for non-member users
198             WestCrossChessClubUser chessClubUser = new WestCrossChessClubUser(session);
199             chessClubUser.setVisible(true);
200             this.dispose();            
201         }else if(session.getIsClubMember() && session.getIsClubLeader()==false){//takes the user to the chess club form for member users
202             WestCrossChessClubMember chessClubMember = new WestCrossChessClubMember(session);
203             chessClubMember.setVisible(true);
204             this.dispose();             
205         }else{//takes the user to the chess club form for the chess club leader
206             WestCrossChessClubLeader chessClubLeader = new WestCrossChessClubLeader(session);
207             chessClubLeader.setVisible(true);
208             this.dispose();   
209         }
