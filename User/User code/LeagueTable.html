 52     /**
 53      * Sets the league table based on the league table data
 54      * @param leagueTableData The league table data
 55      */
 56     private void setTable(String[][]leagueTableData){
 57         String[][]leagueDataFormatted=new String[leagueTableData.length][6];//stores the formatted data, which is used to display the table
 58         int pos=1; //stores the position
 59         String prevPoints="";//stores the points tally of the player directly above a certain user
 60         for(int player=0;player<leagueTableData.length;player++){//loops through every player in the league table and calculates their position in the table. Players who have the same number of points are assigned the same position in the table
 61             if(player==0){
 62                 prevPoints=String.valueOf(leagueTableData[player][4]);
 63                 //copies various attributes from the league table data in the appropriate format
 64                 leagueDataFormatted[player][0]=String.valueOf(pos);
 65                 leagueDataFormatted[player][1]=leagueTableData[player][0];
 66                 leagueDataFormatted[player][2]=leagueTableData[player][1];leagueDataFormatted[player][3]=leagueTableData[player][2];
 67                 leagueDataFormatted[player][4]=leagueTableData[player][3];leagueDataFormatted[player][5]=leagueTableData[player][4];
 68             }else{
 69                 if(!prevPoints.equals(leagueTableData[player][4])){
 70                     pos++;//increments the position if the points of this player are not the same as the points of the previous player
 71                 }
 72                 prevPoints=String.valueOf(leagueTableData[player][4]);
 73                 leagueDataFormatted[player][0]=String.valueOf(pos);
 74                 //copies various attributes from the league table data in the appropriate format
 75                 leagueDataFormatted[player][1]=leagueTableData[player][0];
 76                 leagueDataFormatted[player][2]=leagueTableData[player][1];leagueDataFormatted[player][3]=leagueTableData[player][2];
 77                 leagueDataFormatted[player][4]=leagueTableData[player][3];leagueDataFormatted[player][5]=leagueTableData[player][4];
 78             }
 79         }
 80         JTable tab = new JTable(leagueDataFormatted, COLUMN_NAMES);
 81         leagueTable.setModel(tab.getModel());//sets the league table to the formatted data
 82     }
 83     /**
 84      * Sets the league table based on the members of the club
 85      * @param allClubMembers The club members
 86      */
 87     private void setTable(String[]allClubMembers){
 88         //this method is called when the fixtures have not yet been generated. Hence, the table just displays all users as being in first place and having
 89         //0 points
 90         String [][]tableData=new String[allClubMembers.length][6];
 91         for(int i=0;i<tableData.length;i++){
 92             tableData[i][0]="1";tableData[i][1]=allClubMembers[i];
 93             tableData[i][2]="0";tableData[i][3]="0";tableData[i][4]="0";
 94             tableData[i][5]="0";
 95         }
 96         JTable tab = new JTable(tableData, COLUMN_NAMES);
 97         leagueTable.setModel(tab.getModel());//sets the table to the formatted value
 98     }
