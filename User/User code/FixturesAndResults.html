187     /**
188      * Updates the table based on the data stored in the object
189      */
190     private void updateTable(){
191         if(haveFixturesBeenGenerated){//displays the fixtures and results
192             roundLbl.setText("Round "+String.valueOf(roundIndex+1));//dispalys which round is being viewed
193             //enables/disables the buttons which take the user to the next and previous round depending on which roudn is being accessed
194             prevRoundBtn.setVisible(true);
195             prevRoundBtn.setEnabled(roundIndex!=0);
196             nextRoundBtn.setEnabled(roundIndex!=(fixturesAndResults.length-1));
197             nextRoundBtn.setVisible(true);
198             //sets the fixtures and results table to be visible
199             jScrollPane1.setEnabled(true);jScrollPane1.setVisible(true);
200             fixturesAndResultsTbl.setEnabled(true);fixturesAndResultsTbl.setVisible(true);
201             //sets table
202             String[][]tableData=new String[fixturesAndResults[0][0].length][COLUMN_NAMES.length];
203             //in the form: white, black, result
204             for(int player=0;player<tableData.length;player++){
205                 tableData[player][0]=fixturesAndResults[roundIndex][0][player];
206                 if(tableData[player][0].equals(Fixture.BY_NAME)){//replaces a bye player with an empty value
207                     tableData[player][0]="";
208                 }
209                 tableData[player][1]=fixturesAndResults[roundIndex][1][player];
210                 if(tableData[player][1].equals(Fixture.BY_NAME)){
211                     tableData[player][1]="";
212                 }
213                 //converts the game state to a value that makes sense to the member's viewing the table
214                 switch (fixturesAndResults[roundIndex][2][player]) {
215                     case Fixture.BYE_GAME_STATE:
216                         tableData[player][2]="Bye Game";
217                         break;
218                     case Fixture.GAME_DRAW:
219                         tableData[player][2]="Draw";
220                         break;
221                     case Fixture.GAME_WHITE_WIN:
222                         tableData[player][2]="White Win";
223                         break;
224                     case Fixture.GAME_BLACK_WIN:
225                         tableData[player][2]="Black Win";
226                         break;
227                     default:
228                         //game has not yet been played
229                         tableData[player][2]="Game Not Yet Played";
230                         break;
231                 }
232             }
233             //sets the table to the data generated
234             JTable tab = new JTable(tableData, COLUMN_NAMES);
235             fixturesAndResultsTbl.setModel(tab.getModel());
236             return;
237         }else{
238             //displays that the fixtures have not been generated and updates the form's components accordingly
239             roundLbl.setText("The fixtures have not yet been generated");
240             prevRoundBtn.setEnabled(false);prevRoundBtn.setVisible(false);
241             nextRoundBtn.setEnabled(false);nextRoundBtn.setVisible(false);
242             fixturesAndResultsTbl.setEnabled(false);fixturesAndResultsTbl.setVisible(false);
243             jScrollPane1.setEnabled(false);jScrollPane1.setVisible(false);
244             return;
245         }
